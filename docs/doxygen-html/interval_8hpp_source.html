21  /// Type of each element on the stack used by integral() and by class
22  /// interpolant. An instance of interval stores the value of a function's
23  /// argument and of the function's return value at the end points of an
24  /// interval.
25  /// \tparam A Type of argument to function.
26  /// \tparam R Type returned by function.
27  template <typename A, typename R>
28  struct interval {
29  A a; ///< Function's argument at left end of interval.

32  R fb; ///< Function's return value at right end of interval.
33  };

35  /// Stack used by integral() and by class interpolant.
36  ///
37  /// On construction, an instance of interval_stack contains a partition of
38  /// an interval into subintervals of equal length. As integral() or one of
39  /// interpolant's constructors executes, however, each of these subintervals
40  /// is adaptively subdivided as necessary to achieve the desired accuracy.
41  ///
42  /// The instance of interval_stack is constructed so that the largest
43  /// function values are at the top of the stack (the back of the vector).
44  /// This allows the adaptive-quadrature integrator, which starts at the top
45  /// of the stack, to minimize the time spent in subdividing. If, on
46  /// construction, the top of the stack had an interval corresponding to a
47  /// region where the function had smaller absolute values, then the
48  /// integrator would spend time subidividing for a relative precision that
49  /// would be thrown away when larger function absolute values are
50  /// encountered farther down the stack.
51  ///
52  /// \tparam A Type of argument to function.
53  /// \tparam R Type returned by function.
54  template <typename A, typename R>
55  struct subinterval_stack : public std::vector<interval<A, R>> {
56 
57  using I = interval<A, R>; ///< Type of interval on stack.
58 
59  /// Allow sorting of intervals by function value.
60  static bool compare(I const &iv1, I const &iv2)
61  {
62  R const a1 = fabs(iv1.fa);
63  R const a2 = fabs(iv2.fa);
64  R const b1 = fabs(iv1.fb);
65  R const b2 = fabs(iv2.fb);
66  if (a1 > b1) {
67  if (a2 > b2) {
68  return a1 < a2;
69  } else {
70  return a1 < b2;
71  }
72  } else {
73  if (a2 > b2) {
74  return b1 < a2;
75  } else {
76  return b1 < b2;
77  }
78  }
79  };

81  /// Initialize stack on construction.
82  ///
83  /// Evaluate function at each of n equally spaced points across the
84  /// interval. The function is evaluated at a minimum of two points (the
85  /// end points). The n-1 corresponding instances of interval are pushed
86  /// onto the stack.
87  ///
88  /// \tparam F Type of function.
89  /// \param n Initial number of points at which function is evaluated.
90  /// \param a Starting value of function's argument.
91  /// \param b Ending value of function's argument.
92  /// \param f Function.
93  template <typename F>
94  subinterval_stack(unsigned n, A a, A b, F &&f)
95  {
96  if (n < 2) {
97  n = 2;
98  }
99  A const d = (b - a) / (n - 1);
100  A ta = a;
101  // Loop starts at 1 (not zero) to ensure n-1 iterations (not n).
102  for (unsigned j = 1; j < n; ++j) {
103  A const tb = ta + d;
104  this->push_back({ta, tb, f(ta), f(tb)});
105  ta = tb;
106  }
107  sort(this->begin(), this->end(), compare);
108  }
109  };
110 }
