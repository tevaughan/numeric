15  /// Base class for model of a function of a continuous variable. The
16  /// immediate descendant's type \a D is a template type parameter. See
17  /// [CRTP](https://en.wikipedia.org/wiki/Curiously_recurring_template_pattern).
18  ///
19  /// The descendant may use virtual functions for its descendants but is not
20  /// required to, and so we may pay the cost of dynamic polymorphism on a
21  /// per-descendant basis.
22  ///
23  /// \tparam A Type of function's argument.
24  /// \tparam R Type returned by function.
25  /// \tparam D Type of descendant.
26  template <typename A, typename R, typename D>
27  struct cfunc {
28  /// Type of derivative.
29  using DERIV = decltype(D().derivative());
30 
31  /// Type of indefinite integral.
32  using INTEG = decltype(D().integral());
33 
34  /// Return pointer to descendant instance.
35  D const &d() const { return static_cast<D const *>(this); }
36 
37  /// Evaluate function.
38  R operator()(A const& a) const { return d()(a); }
39 
40  /// Return function representing derivative.
41  DERIV derivative() const { return d()->derivative(); }
42 
43  /// Return function representing integral from specified lower bound.
44  INTEG integral(A const &lb) const { return d()->integral(lb); }
45  };
