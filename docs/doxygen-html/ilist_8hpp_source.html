60  /// Extract point from space-delimited ASCII line.
61  /// \tparam X Type of first column, representing X coordinate.
62  /// \tparam Y Type of second column, representing Y coordinate.
63  template <typename X, typename Y>
64  ipoint<X, Y> get_point(
65  /** Line of non-blank ASCII text. */ std::string line,
66  /** Unit to multiply against first column. */ X xu,
67  /** Unit to multiply against secnd column. */ Y yu)

77  return ipoint<X, Y>(x * xu, y * yu);

80  /// Extract a set of points from a space-delimited ASCII file. Each line
81  /// of the file must consist either
82  ///
83  /// - of only white space, optionally followed by the '#' character and a
84  /// subsequent comment or
85  ///
86  /// - of white space followed by at least two space-delimited
87  /// floating-point numbers (after which everything else on the line is
88  /// ignored).
89  ///
90  /// \tparam X Type of first column, representing X coordinate.
91  /// \tparam Y Type of second column, representing Y coordinate.
92  template <typename X, typename Y>
93  ilist<X, Y> get_points(
94  /** Name of ASCII file. */ std::string file,
95  /** Unit to multiply against first column. */ X xu,
96  /** Unit to multiply against secnd column. */ Y yu)
97  {
98  std::ifstream is(file);
99  if (!is) {
100  throw "Failed to open '" + file + "'.";
101  }
102  std::string line;
103  ilist<X, Y> points;
104  while (getline(is, line)) {
105  size_t const p = line.find_first_not_of(" \f\n\r\t");
106  if (p == std::string::npos || line[p] == '#') {
107  continue;
108  }
109  points.push_back(get_point(line, xu, yu));
110  }
111  auto comp = [](ipoint<X, Y> const &a, ipoint<X, Y> const &b) {
112  return a.first < b.first;
113  };
114  sort(points.begin(), points.end(), comp);
115  return points;
116  }

118  /// Return the list of midpoints, each between a subsequent pair of
119  /// points in a sorted list.
120  ///
121  /// \tparam V Type of vector of points.
122  /// \return Midpoints sorted by x coordinate.
123  template <typename V>
124  V midpoints(/** Points sorted by x coordinate. */ V const &v)
125  {
126  if (v.size() < 2) {
127  return V();
128  }
129  V rv(v.size() - 1); // Return value.
130  for (unsigned j = 1; j < v.size(); ++j) {
131  unsigned const i = j - 1;
132  auto const & xi = v[i].first, &yi = v[i].second;
133  auto const & xj = v[j].first, &yj = v[j].second;
134  rv[i] = {0.5 * (xi + xj), 0.5 * (yi + yj)};
135  }
136  return rv;
137  }
138 }
