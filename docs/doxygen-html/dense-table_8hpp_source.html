12  /// Simple, constant-time-lookup model of a function of a continuous
13  /// variable.
14  ///
15  /// dense_table stores a function's value at each point on a regular grid
16  /// across the value of the function's argument. So each pair of subsequent
17  /// table entries is separated by the same distance \f$\Delta a\f$ in the
18  /// space of the function's argument.
19  ///
20  /// For log-time-lookup, see sparse_table.
21  ///
22  /// \tparam A Type of function's argument.
23  /// \tparam R Type of function's return value.
24  template<typename A, typename R>
25  class dense_table
26  {
27  using I = decltype(1.0 / A());
28  A a0_; ///< First tabulated argument.
29  A a_min_; ///< Minimum argument for tabulated value.
30  A a_max_; ///< Maximum argument for tabulated value.
31  A da_; ///< Difference between subsequent tabulated args.
32  I ida_; ///< Inverse of difference between subsequent tabulated args.
33  std::vector<R> r_; ///< Tabulated return values.
34 
35  public:
36  /// Initialize table from the first tabulated argument, the difference
37  /// between subsequent arguments, and the list of tabulated return
38  /// values.

42  )
43  : a0_(a)
44  , a_min_(a - 0.5 * d)
