12  /// Simple, constant-time-lookup model of a function of a continuous
13  /// variable.
14  ///
15  /// To enable constant-time lookup, dense_table contains a regular grid
16  /// across the value of the function's argument. So each pair of subsequent
17  /// table entries is separated by the same distance \f$ \Delta a \f$ in the
18  /// space of the function's argument.
19  ///
20  /// At the \f$ i^{\text{th}} \f$ entry, corresponding to the value \f$ a_i
21  /// \f$ of the function's argument, an instance of dense_table stores a
22  /// function object \f$ f_i \f$ of type \a F.
23  ///
24  /// When the lookup occurs, by way of dense_table::operator()(A const&), let
25  /// the value of the passed-in argument be \f$ a \f$. When \f$ a \f$ is
26  /// neither before the first table entry nor after the last, the index \f$ i
27  /// \f$ is chosen so that \f[ -\frac{\Delta a}{2} < a - a_i \leq
28  /// +\frac{\Delta a}{2}. \f] What is returned by dense_table::operator() is
29  /// the value \f$ f_i(a - a_i) \f$.
30  ///
31  /// For log-time-lookup, see sparse_table.
32  ///
33  /// \tparam A Type of function's argument.
34  ///
35  /// \tparam F Type of each function \f$ f_i \f$, of which an instance of
36  /// dense_table is a piece-wise construction.
37  template<typename A, typename F>
38  class dense_table
39  {
40  using I = decltype(1.0 / A());
41  A a_frst_; ///< First tabulated argument.
42  A da_; ///< Difference between subsequent tabulated args.
43  I ida_; ///< Inverse of difference between subsequent tabulated args.
44  std::vector<F> f_; ///< Tabulated return values.
45 
46  public:
47  /// Initialize table from the first tabulated argument, the difference
48  /// between subsequent arguments, and the list of tabulated function
49  /// objects.

52  std::vector<F> vf ///< Tabulated function objects.

72 
73  /// Value returned by tabulated function object associated with supplied
74  /// argument.
75  auto operator()(A const& a /**< Desired argument. */) const -> decltype(F()(a))
76  {
77  if (a < a_frst_) {
78  return f_[0](a - a_frst_);
79  }
80  A const last = a_last();
81  if (a > last) {
82  return (*f_.rbegin())(a - last);
83  }
84  int const i((a - a_frst_) * ida_ + 0.5);
85  A const ai = a_frst_ + i * da_;
86  return f_[i](a - ai);
87  }
