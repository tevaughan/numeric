17  /// Base class for model of a function of a continuous variable. The
18  /// immediate descendant's type is a template type parameter. See
19  /// [CRTP](https://en.wikipedia.org/wiki/Curiously_recurring_template_pattern).
20  ///
21  /// The descendant may use virtual functions but is not required to, and so
22  /// we may pay the cost of dynamic polymorphism on a per-descendant basis.
23  ///
24  /// \tparam A Type of function's argument.
25  /// \tparam D Type of descendant.
26  template <typename A, typename D>
27  struct rfunc {
28  /// Type of derivative.
29  using DERIV = decltype(D().derivative());
30 
31  /// Type of indefinite integral.
32  using INTEG = decltype(D().integral());
33 
34  /// Return pointer to descendant instance.
35  D const& d() const { return static_cast<D const*>(this); }
36 
37  /// Return function representing derivative.
38  DERIV derivative() const { return d()->derivative(); }
39 
40  /// Return function representing integral from specified lower bound.
41  INTEG integral(A const& lb) const { return d()->integral(lb); }
42  };
43 }
