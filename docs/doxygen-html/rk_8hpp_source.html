133  /// Given the value for variable \a y and the value for its derivative \a
134  /// dydx, use the fifth-order Cash-Karp Runge-Kutta method to advance the
135  /// solution for \a y over an interval \a h, and return the incremented
136  /// variable as \a yout. Also, return, by way of the embedded
137  /// fourth-order method, an estimate \a yerr of the local truncation
138  /// error in \a yout. The supplied function \a deriv returns the
139  /// derivative (just like dydx) at any value of the independent variable.
140  ///
141  /// This function is based on a simplified version of `rkck()` found on
142  /// Page 719 in Numerical Recipes in C, Second Edition. The
143  /// simplification is due mainly to the fact that the present function is
144  /// used for quadrature, and so \a deriv does not require \a y as input.
145  ///
146  void
147  rkck(/** Length of interval. */ X const &h,
148  /** Accumulated value at end of interval. */ Y & out,
149  /** Estimate of local truncation error. */ Y & e)
150  {
151  // 1st step is given as dydx on input.
152  // 2nd step not needed because deriv() does not need y as input.
153  X const x3 = x + a3 * h;
154  X const x4 = x + a4 * h;
155  X const x5 = x + a5 * h;
156  X const x6 = x + a6 * h;
157  RAT<Y, X> const ak3 = deriv(x3); // 3rd step.
158  RAT<Y, X> const ak4 = deriv(x4); // 4th step.
159  RAT<Y, X> const ak5 = deriv(x5); // 5th step.
160  RAT<Y, X> const ak6 = deriv(x6); // 6th step.
161  // Accumulate increments with proper weights.
162  out = y + h * (c1 * dydx + c3 * ak3 + c4 * ak4 + c6 * ak6);
163  // Estimate eor as difference between fourth- and fifth-order
164  // methods.
165  e = h * (dc1 * dydx + dc3 * ak3 + dc4 * ak4 + dc5 * ak5 + dc6 * ak6);
166  }

168  /// See implementation of `rkqs()` on Paqe 719 in Numerical Recipes in C,
169  /// Second Edition.
170  static double constexpr SAFETY = 0.9;

172  /// Reduce stepsize no more than a factor of 10.
173  /// \return New stepsize.
174  static X reduce_step_size(
175  /** Old stepsize. */ X h, /** Truncation error. */ double err)
176  {
177  static double constexpr PSHRNK = -0.25;
178  X const htemp = SAFETY * h * pow(err, PSHRNK);
179  X const tenth = 0.1 * h;
180  static X const zero = 0.0 * h;
181  if (h >= zero) {
182  if (htemp > tenth) {
183  h = htemp;
184  } else {
185  h = tenth;
186  }
187  } else {
188  if (htemp < tenth) {
189  h = htemp;
190  } else {
191  h = tenth;
192  }
193  }
194  return h;
195  }

197  /// Fifth-order Runge-Kutta step with monitoring of local truncation
198  /// error to ensure accurcy and adjust stepsize. Input are the
199  /// accumulated variable \a y and its derivative \a dydx at the starting
200  /// value of the independent variable \a x. Also input are the stepsize
201  /// \a htry to be attempted and the required accuracy \a tol. On output,
202  /// \a y and \a x are replaced by their new values, \a hdid is the
203  /// stepsize that was actually accomplished, and \a hnext is the
204  /// estimated next stepsize. The supplied function \a deriv returns the
205  /// derivative (just like dydx) at any value of the independent variable.
206  ///
207  /// This function is based on `rkqs()` found on Page 719 in Numerical
208  /// Recipes in C, Second Edition.
209  ///
210  void
211  rkqs(/** Stepsize to be attempted. */ X const &htry,
212  /** Scaling used to monitor accuracy. */ Y const &yscal,
213  /** Stepsize that was accomplished. */ X & hdid,
214  /** Estimated next stepsize. */ X & hnext)
215  {
216  double err;
217  Y yerr;
218  Y ytemp;
219  X h = htry; // initial trial value
220  while (true) {
221  rkck(h, ytemp, yerr); // Take a trial step.
222  err = fabs(yerr / yscal / tol);
223  if (err <= 1.0) {
224  break;
225  }
226  // Truncation error is too large.
227  h = reduce_step_size(h, err);
228  if (x + h == x) {
229  throw "stepsize underflow";
230  }
231  }
232  // Increase stepsize no more than a factor of 5.
233  static double constexpr PGROW = -0.2;
234  static double const ERRCON = pow(5.0 / SAFETY, 1.0 / PGROW);
235  if (err > ERRCON) {
236  hnext = SAFETY * h * pow(err, PGROW);
237  } else {
238  hnext = 5.0 * h;
239  }
240  x += (hdid = h);
241  y = ytemp;
242  }

256  /// Make sure that n is reasonable, and pick initial guess at stepsize.
257  X initial_h(
258  /** Lower limit of integration. */ X const &x1,
259  /** Upper limit of integration. */ X const &x2,
260  /** Number of equal-size steps. */ int & n)
261  {
262  if (n < 2) {
263  n = 2;
264  }
265  X const h1 = (X(x2) - X(x1)) / (n - 1);
266  if (x2 - x1 > 0.0 * x2) {
267  return +fabs(h1);
268  } else {
269  return -fabs(h1);
270  }
271  }

273  /// This function is based on `odeint()` on Page 721 of Numerical Recipes
274  /// in C, Second Edition.
275  void
276  init(/** Lower limit of integration. */ X x1,
277  /** Upper limit of integration. */ X x2,
278  /** Number of equal-size steps. */ int n)
279  {
280  check_tol();
281  X h = initial_h(x1, x2, n);
282  using namespace std;
283  static const PRD<X, X> XSQR_0 = 0.0 * x1 * x1;
284  while (true) {
285  dydx = deriv(x);
286  // This doesn't work when Y is dyndim.
287  static Y const TINY = tiny<Y>::val(dydx * h);
288  // General-purpose scaling used to monitor accuracy.
289  Y const yscal = fabs(y) + fabs(dydx * h) + TINY;
290  if (store) {
291  dl.push_back({x, dydx});
292  yl.push_back({x, y}); // y=0 on first time through loop.
293  }
294  X const xh = x + h;
295  if ((xh - x2) * (xh - x1) > XSQR_0) {
296  h = x2 - x; // Decrease stepsize to avoid overshoot.
297  }
298  X hdid, hnext;
299  rkqs(h, yscal, hdid, hnext);
300  if (hdid == h) {
301  ++nok;
302  } else {
303  ++nbad;
304  }
305  if ((x - x2) * (x2 - x1) >= XSQR_0) {
306  if (store) {
307  dl.push_back({x, deriv(x)});
308  yl.push_back({x, y});
309  }
310  return; // We are done; exit normally.
311  }
312  if (fabs(hnext) <= 0.0 * hnext) {
313  cerr << "rk_quad: WARNING: step size too small" << endl;
314  return;
315  }
316  h = hnext;
317  }
318  cerr << "rk_quad: WARNING: too many steps" << endl;
319  }

405  /// Construct the quadratic interpolant through the integrated function's
406  /// evaluated points (the same points as returned by intermed_fnc()), so
407  /// that the curvature of each piece is adjusted for agreement with the
408  /// Runge-Kutta estimate (returned by intermed_int()) of the integral
409  /// over the same piece.
410  ///
411  /// The coefficients of a quadratic interpolant between two points can be
412  /// determined if the area under the interpolant be known. For each piece
413  /// between subsequent points
414  /// \f[
415  /// p_i = (x_i,y_i)
416  /// \f]
417  /// \f[
418  /// p_{i+1} = (x_{i+1},y_{i+1})
419  /// \f]
420  /// returned by intermed_fnc(), there are two corresponding points
421  /// \f$(x_i,Y_i)\f$ and \f$(x_{i+1},Y_{i+1})\f$ returned by
422  /// intermed_int(). (The notation used here is different from that used
423  /// in the documentation for intermed_int().) The difference,
424  /// \f[
425  /// A_i = Y_{i+1} - Y_i
426  /// \f]
427  /// is the Runge-Kutta estimate of the area under the function that
428  /// passes through \f$p_i\f$ and \f$p_{i+1}\f$.
429  ///
430  /// The linear interpolant between the points has the form
431  /// \f[
432  /// \lambda_i(x) = y_i + m_i [x - x_i],
433  /// \f]
434  /// where the slope is
435  /// \f[
436  /// m_i = \frac{\Delta y_i}{\Delta x_i};
437  /// \f]
438  /// the area under the line is
439  /// \f[
440  /// a_i = \Delta x_i \; \left[ y_i + \frac{\Delta y_i}{2} \right];
441  /// \f]
442  /// and where
443  /// \f[
444  /// \Delta x_i = x_{i+1} - x_i
445  /// \f]
446  /// \f[
447  /// \Delta y_i = y_{i+1} - y_i.
448  /// \f]
449  ///
450  /// The quadratic interpolant that not only passes through \f$p_i\f$ and
451  /// \f$p_{i+1}\f$ but also covers the area \f$A_i\f$ is of the form
452  /// \f[
453  /// q_i(x) = y_i + [m_i + c_i[x - x_{i+1}]][x - x_i],
454  /// \f]
455  /// where
456  /// \f[
457  /// c_i = -\frac{6 [A_i - a_i]}{[\Delta x_i]^3}.
458  /// \f]
459  sparse_table<X> make_fnc_interp() const
460  {
461  if (dl.size() < 2) {
462  throw "Must have at least two control points.";
463  }
464  // Each subdomain is the x region between subsequent control points.
465  X const a0 = 0.5 * (dl[0].first + dl[1].first);
466  unsigned const nd = dl.size() - 1; // number of deltas
467  using namespace std;
468  vector<pair<X, GiNaC::ex>> vf(nd);
469  for (unsigned i = 0; i < nd; ++i) {
470  unsigned const j = i + 1;
471  // horizontal geometry
472  X const &x1 = dl[i].first; // left edge of piece
473  X const &x2 = dl[j].first; // right edge of piece
474  X const dx = x2 - x1; // width of piece
475  // vertical geometry
476  DYDX const &y1 = dl[i].second; // func val at left
477  DYDX const &y2 = dl[j].second; // func val at right
478  DYDX const dy = y2 - y1; // change in func val
479  // areas
480  Y const a1 = (y1 + 0.5 * dy) * dx; // under linear interp
481  Y const a2 = yl[j].second - yl[i].second; // of R-K estimate
482  Y const da = a2 - a1; // difference
483  // calculation of coefficients
484  auto const dx3 = dx * dx * dx;
485  auto const c = -6.0 * da / dx3; // quadratic coef
486  auto const m = dy / dx; // slope of linear interp
487  // construction of expression for piece
488  auto const &x = sparse_table<X>::x; // independent var's symbol
489  vf[i].first = dx; // width of piece
490  vf[i].second = y1 + (m + c * (x - x2)) * (x - x1); // expression
491  }
492  return sparse_table<X>(a0, move(vf));
493  }

495  /// Construct the cubic interpolant through the integral's partial-sum
496  /// points (the same points as returned by intermed_int()), so that the
497  /// curvature of each piece is adjusted for agreement with the derivative
498  /// values (returned by intermed_fnc()) at the endpoints of the same
499  /// piece.
500  ///
501  /// The coefficients of a cubic interpolant between two points can be
502  /// determined if the derivative be known at each piece's endpoint. For
503  /// each piece between subsequent points
504  /// \f[
505  /// p_i = (x_i,y_i)
506  /// \f]
507  /// \f[
508  /// p_{i+1} = (x_{i+1},y_{i+1})
509  /// \f]
510  /// returned by intermed_int(), there are two corresponding derivative
511  /// constraints \f$(x_i,y'_i)\f$ and \f$(x_{i+1},y'_{i+1})\f$ returned by
512  /// intermed_fnc(). (The notation used here is different from that used
513  /// in the documentation for intermed_fnc().)
514  ///
515  /// The linear interpolant between the points has the form
516  /// \f[
517  /// \lambda_i(x) = y_i + m_i [x - x_i],
518  /// \f]
519  /// where the slope is
520  /// \f[
521  /// m_i = \frac{\Delta y_i}{\Delta x_i},
522  /// \f]
523  /// and where
524  /// \f[
525  /// \Delta x_i = x_{i+1} - x_i
526  /// \f]
527  /// \f[
528  /// \Delta y_i = y_{i+1} - y_i.
529  /// \f]
530  ///
531  /// The cubic interpolant that not only passes through \f$p_i\f$ and
532  /// \f$p_{i+1}\f$ but also obeys the constraints on the derivative
533  /// returned by intermed_fnc() has the form
534  /// \f[
535  /// \kappa_i(x) = y_i + y'_i [x - x_i] + \frac{3m_i - 2y'_i -
536  /// y'_{i+1}}{\Delta x_i} [x - x_i]^2 + \frac{y'_i + y'_{i+1} -
537  /// 2m_i}{[\Delta x_i]^2} [x - x_i]^3.
538  /// \f]
539  sparse_table<X> make_int_interp() const
540  {
541  if (dl.size() < 2) {
542  throw "Must have at least two control points.";
543  }
544  // Each subdomain is the x region between subsequent control points.
545  X const a0 = 0.5 * (dl[0].first + dl[1].first);
546  unsigned const nd = dl.size() - 1; // number of deltas
547  using namespace std;
548  vector<pair<X, GiNaC::ex>> vf(nd);
549  for (unsigned i = 0; i < nd; ++i) {
550  unsigned const j = i + 1;
551  // horizontal geometry
552  X const &x1 = yl[i].first; // left edge of piece
553  X const &x2 = yl[j].first; // right edge of piece
554  X const dx = x2 - x1; // width of piece
555  // vertical geometry
556  Y const &y1 = yl[i].second; // func val at left
557  Y const &y2 = yl[j].second; // func val at right
558  Y const dy = y2 - y1; // change in func val
559  // derivatives
560  DYDX const &yp1 = dl[i].second; // derivative at left edge
561  DYDX const &yp2 = dl[j].second; // derivative at right edge
562  // calculation of coefficients
563  auto const dx2 = dx * dx;
564  auto const m = dy / dx; // slope of linear interp
565  auto const c = (3.0 * m - 2.0 * yp1 - yp2) / dx; // quadratic coef
566  auto const d = (yp1 + yp2 - 2.0 * m) / dx2; // cubic coef
567  // construction of expression for piece
568  auto const &x = sparse_table<X>::x; // independent var's symbol
569  auto const xt = x - x1; // change of variable
570  vf[i].first = dx; // width of piece
571  vf[i].second = y1 + yp1 * xt + c * pow(xt, 2) + d * pow(xt, 3);
572  }
573  return sparse_table<X>(a0, move(vf));
574  }
