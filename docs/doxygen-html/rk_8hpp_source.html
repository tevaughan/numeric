109  /// Given the value for variable \a y and the value for its derivative \a
110  /// dydx, use the fifth-order Cash-Karp Runge-Kutta method to advance the
111  /// solution for \a y over an interval \a h, and return the incremented
112  /// variable as \a yout. Also, return, by way of the embedded
113  /// fourth-order method, an estimate \a yerr of the local truncation
114  /// error in \a yout. The supplied function \a deriv returns the
115  /// derivative (just like dydx) at any value of the independent variable.
116  ///
117  /// This function is based on a simplified version of `rkck()` found on
118  /// Page 719 in Numerical Recipes in C, Second Edition. The
119  /// simplification is due mainly to the fact that the present function is
120  /// used for quadrature, and so \a deriv does not require \a y as input.
121  ///
122  void
123  rkck(/** Length of interval. */ X const &h,
124  /** Accumulated value at end of interval. */ Y & out,
125  /** Estimate of local truncation error. */ Y & e)
126  {
127  // 1st step is given as dydx on input.
128  // 2nd step not needed because deriv() does not need y as input.
129  X const x3 = x + a3 * h;
130  X const x4 = x + a4 * h;
131  X const x5 = x + a5 * h;
132  X const x6 = x + a6 * h;
133  RAT<Y, X> const ak3 = deriv(x3); // 3rd step.
134  RAT<Y, X> const ak4 = deriv(x4); // 4th step.
135  RAT<Y, X> const ak5 = deriv(x5); // 5th step.
136  RAT<Y, X> const ak6 = deriv(x6); // 6th step.
137  // Accumulate increments with proper weights.
138  out = y + h * (c1 * dydx + c3 * ak3 + c4 * ak4 + c6 * ak6);
139  // Estimate eor as difference between fourth- and fifth-order
140  // methods.
141  e = h * (dc1 * dydx + dc3 * ak3 + dc4 * ak4 + dc5 * ak5 + dc6 * ak6);
142  }

144  /// See implementation of `rkqs()` on Paqe 719 in Numerical Recipes in C,
145  /// Second Edition.
146  static double constexpr SAFETY = 0.9;
147 
148  /// Reduce stepsize no more than a factor of 10.
149  /// \return New stepsize.
150  static X reduce_step_size(
151  /** Old stepsize. */ X h, /** Truncation error. */ double err)
152  {
153  static double constexpr PSHRNK = -0.25;
154  X const htemp = SAFETY * h * pow(err, PSHRNK);
155  X const tenth = 0.1 * h;
156  static X const zero(0.0);
157  if (h >= zero) {
158  if (htemp > tenth) {
159  h = htemp;
160  } else {
161  h = tenth;
162  }
163  } else {
164  if (htemp < tenth) {
165  h = htemp;
166  } else {
167  h = tenth;
168  }
169  }
170  return h;
171  }

173  /// Fifth-order Runge-Kutta step with monitoring of local truncation
174  /// error to ensure accurcy and adjust stepsize. Input are the
175  /// accumulated variable \a y and its derivative \a dydx at the starting
176  /// value of the independent variable \a x. Also input are the stepsize
177  /// \a htry to be attempted and the required accuracy \a tol. On output,
178  /// \a y and \a x are replaced by their new values, \a hdid is the
179  /// stepsize that was actually accomplished, and \a hnext is the
180  /// estimated next stepsize. The supplied function \a deriv returns the
181  /// derivative (just like dydx) at any value of the independent variable.
182  ///
183  /// This function is based on `rkqs()` found on Page 719 in Numerical
184  /// Recipes in C, Second Edition.
185  ///
186  void
187  rkqs(/** Stepsize to be attempted. */ X const &htry,
188  /** Scaling used to monitor accuracy. */ Y const &yscal,
189  /** Stepsize that was accomplished. */ X & hdid,
190  /** Estimated next stepsize. */ X & hnext)
191  {
192  double err;
193  Y yerr;
194  Y ytemp;
195  X h = htry; // initial trial value
196  while (true) {
197  rkck(h, ytemp, yerr); // Take a trial step.
198  err = fabs(yerr / yscal / tol);
199  if (err <= 1.0) {
200  break;
201  }
202  // Truncation error is too large.
203  h = reduce_step_size(h, err);
204  if (x + h == x) {
205  throw "stepsize underflow";
206  }
207  }
208  // Increase stepsize no more than a factor of 5.
209  static double constexpr PGROW = -0.2;
210  static double const ERRCON = pow(5.0 / SAFETY, 1.0 / PGROW);
211  if (err > ERRCON) {
212  hnext = SAFETY * h * pow(err, PGROW);
213  } else {
214  hnext = 5.0 * h;
215  }
216  x += (hdid = h);
217  y = ytemp;
218  }

220  /// Make sure that tolerance is neither negative nor too small.
221  void check_tol()
222  {
223  double constexpr eps = std::numeric_limits<double>::epsilon();
224  double constexpr min_tol = 100.0 * eps;
225  if (tol <= 0.0) {
226  throw "tolerance not positive";
227  } else if (tol < min_tol) {
228  tol = min_tol;
229  }
230  }

232  /// Make sure that n is reasonable, and pick initial guess at stepsize.
233  X initial_h(
234  /** Lower limit of integration. */ X const &x1,
235  /** Upper limit of integration. */ X const &x2,
236  /** Number of equal-size steps. */ int & n)
237  {
238  if (n < 2) {
239  n = 2;
240  }
241  X const h1 = (X(x2) - X(x1)) / (n - 1);
242  if (x2 - x1 > X(0.0)) {
243  return +fabs(h1);
244  } else {
245  return -fabs(h1);
246  }
247  }

249  /// This function is based on `odeint()` on Page 721 of Numerical Recipes
250  /// in C, Second Edition.
251  void
252  init(/** Lower limit of integration. */ X x1,
253  /** Upper limit of integration. */ X x2,
254  /** Number of equal-size steps. */ int n)
255  {
256  check_tol();
257  X h = initial_h(x1, x2, n);
258  using namespace std;
259  static const PRD<X, X> XSQR_0 = 0.0;
260  while (true) {
261  dydx = deriv(x);
262  static Y const TINY(1.0E-300);
263  // General-purpose scaling used to monitor accuracy.
264  Y const yscal = fabs(y) + fabs(dydx * h) + TINY;
265  if (store) {
266  dl.push_back({x, dydx});
267  yl.push_back({x, y}); // y=0 on first time through loop.
268  }
269  X const xh = x + h;
270  if ((xh - x2) * (xh - x1) > XSQR_0) {
271  h = x2 - x; // Decrease stepsize to avoid overshoot.
272  }
273  X hdid, hnext;
274  rkqs(h, yscal, hdid, hnext);
275  if (hdid == h) {
276  ++nok;
277  } else {
278  ++nbad;
279  }
280  if ((x - x2) * (x2 - x1) >= XSQR_0) {
281  if (store) {
282  dl.push_back({x, deriv(x)});
283  yl.push_back({x, y});
284  }
285  return; // We are done; exit normally.
286  }
287  if (fabs(hnext) <= 0.0) {
288  cerr << "rk_quad: WARNING: step size too small" << endl;
289  return;
290  }
291  h = hnext;
292  }
293  cerr << "rk_quad: WARNING: too many steps" << endl;
294  }

368  /// Construct quadratic interpolant through intermediate points (same
369  /// points as returned by intermed_fnc()), so that curvature of each
370  /// piece is adjusted for agreement with integral (returned by
371  /// intermed_int()) of the same piece.
372  sparse_table<X> make_fnc_interp() const
373  {
374  if (dl.size() < 2) {
375  throw "Must have at least two control points.";
376  }
377  // Each subdomain is the x region between subsequent control points.
378  X const a0 = 0.5 * (dl[0].first + dl[1].first);
379  unsigned const nd = dl.size() - 1; // number of deltas
380  using namespace std;
381  vector<pair<X, GiNaC::ex>> vf(nd);
382  for (unsigned i = 0; i < nd; ++i) {
383  unsigned const j = i + 1;
384  X const & x1 = dl[i].first;
385  X const & x2 = dl[j].first;
386  X const dx = x2 - x1;
387  DYDX const & y1 = dl[i].second;
388  DYDX const & y2 = dl[j].second;
389  DYDX const dy = y2 - y1;
390  Y const a1 = (y1 + 0.5 * dy) * dx;
391  Y const a2 = yl[j].second - yl[i].second;
392  Y const da = a2 - a1;
393  auto const dx3 = dx * dx * dx;
394  auto const c2 = -6.0 * da / dx3;
395  auto const c1 = dy / dx - c2 * (x1 + x2);
396  DYDX const c0 = y1 - (c1 + c2 * x1) * x1;
397  auto const & x = sparse_table<X>::x;
398  vf[i].first = dx;
399  vf[i].second = c0 + c1 * x + c2 * pow(x, 2);
400  }
401  return sparse_table<X>(a0, move(vf));
402  }
403  };
