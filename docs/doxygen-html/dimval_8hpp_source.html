113  /// Add exponents for multiplication of two dimensioned quantities.
114  dim_exps operator+(dim_exps ode /**< Other exponents. */) const

124  C() - ode.C(), TE() - ode.TE());

127  /// Multiply exponents by integer as for integer power of dimensioned
128  /// value.
129  dim_exps operator*(int p /**< Integer power. */) const
130  {
131  return dim_exps(TI() * p, D() * p, M() * p, C() * p, TE() * p);
132  }
133 
134  /// Divide exponents by integer as for integer root of dimensioned value.
135  dim_exps operator/(int r /**< Integer root. */) const
136  {
137  if (r <= 0) {
138  throw "zero or negative root";
139  }
140  if (D() / r * r != D()) {
141  throw "illegal root along distance";
142  }
143  if (M() / r * r != M()) {
144  throw "illegal root along mass";
145  }
146  if (C() / r * r != C()) {
147  throw "illegal root along charge";
148  }
149  if (TI() / r * r != TI()) {
150  throw "illegal root along time";
151  }
152  if (TE() / r * r != TE()) {
153  throw "illegal root along temperature";
154  }
155  return dim_exps(TI() / r, D() / r, M() / r, C() / r, TE() / r);
156  }
157 
158  /// Return true only if every exponent be same as correspondent in other
159  /// set.
160  bool operator==(dim_exps ode /**< Other exponents. */) const
161  {
162  return n_ == ode.n_;
163  }
164 
165  /// Return true only if any exponent be different from correspondent in
166  /// other set.
167  bool operator!=(dim_exps ode /**< Other exponents. */) const
168  {
169  return n_ != ode.n_;

172  /// Return negative exponents as for reciprocal of dimensioned value.
173  dim_exps neg() const { return dim_exps(-TI(), -D(), -M(), -C(), -TE()); }
