22  /// Model of a polynomial of a continuous variable.
23  ///
24  /// Class cpoly inherits from \ref cfunc because cpoly is a model of a
25  /// continuous function.
26  ///
27  /// \tparam D Degree of polynomial.
28  /// \tparam V Type of variable. By default, primitive type double.
29  /// \tparam C Type of each term. By default, primitive type double.
30  template <unsigned D, typename V = double, typename C = double>
31  class cpoly : public cfunc<V, C, cpoly<D, V, C>>
32  {
33  static unsigned constexpr N = D + 1; ///< Number of coefficients.
34  std::array<C, N> c_; ///< Normalized coefficients.
35 
36  public:
37  /// By default, initialize coefficients to zero.
38  cpoly() { memset(c_.data(), 0, sizeof(c_)); }
39 
40  /// Initialize coefficients.
41  template <typename T>
42  cpoly(std::array<T, N> const &a)
43  {
44  for (unsigned i = 0; i < N; ++i) {
45  c_[i] = a[i] * pow(V(1.0), i);
46  }
47  }
48 
49  /// Initialize coefficients.
50  template <typename T>
51  cpoly(std::vector<T> const &v)
52  {
53  if (v.size() != N) {
54  throw "Wrong number of coefficients.";
55  }
56  for (unsigned i = 0; i < N; ++i) {
57  c_[i] = v[i] * pow(V(1.0), i);
58  }
59  }
60 
61  /// Return number of coefficients in polynomial.
62  static unsigned constexpr num_coefs() { return N; }
63 
64  /// Type of coefficient for term of degree \a I.
65  /// \tparam I Degree of term.
66  template <unsigned I>
67  using ctype = decltype(C(1.0) / pow<I>(V(1.0)));
68 
69  /// Fetch coefficient for term of degree \a I.
70  /// \tparam I Degree of term.
71  template <unsigned I>
72  ctype<I> coef() const
73  {
74  static_assert(I <= D, "Array access must be in bounds.");
75  return c_[I] / pow<I>(V(1.0));
76  }
77 
78  /// Set coefficient for term of degree \a I.
79  /// \tparam I Degree of term.
80  template <unsigned I>
81  void set_coef(ctype<I> const &c)
82  {
83  static_assert(I <= D, "Array access must be in bounds.");
84  c_[I] = c * pow<I>(V(1.0));
85  }
86 
87  /// Evaluate polynomial.
88  C operator()(/** Value of variable in polynomial. */ V const &v) const
89  {
90  double const vn = v / V(1.0); // Normalized value of variable.
91  double vp = vn; // Initial power of normalized value.
92  C val = c_[0];
93  for (unsigned i = 1; i < c_.size(); ++i) {
94  val += c_[i] * vp;
95  vp *= vn;
96  }
97  return val;
98  }
99 
100  using dterm = decltype(C(1.0) / V(1.0)); ///< Type of derivative's term.
101  using iterm = decltype(C(1.0) * V(1.0)); ///< Type of integral's term.
102  using deriv = cpoly<D - 1, V, dterm>; ///< Type of derivative.
103  using integ = cpoly<D + 1, V, iterm>; ///< Type of integral.
104  friend class cpoly<D - 1, V, dterm>; ///< Integral needs access.
105  friend class cpoly<D + 1, V, iterm>; ///< Derivative needs access.
106 
107  /// Return function representing derivative.
108  deriv derivative() const
109  {
110  deriv d;
111  for (unsigned i = 1; i < c_.size(); ++i) {
112  d.c_[i - 1] = i * c_[i] / V(1.0);
113  }
114  return d;
115  }
116 
117  /// Return function representing integral from specified lower bound.
118  integ integral(/** Lower bound of integration. */ V const &lb) const
119  {
120  integ i;
121  // Initialize constant with zero in right units.
122  i.c_[0] = 0.0 * (*this)(lb)*lb;
123  double const u = lb / V(1.0);
124  double lbn = u; // Initialize power of normalized lower bound.
125  for (unsigned j = 0; j < c_.size(); ++j) {
126  iterm const cn = c_[j] / (j + 1) * V(1.0);
127  i.c_[j + 1] = cn;
128  i.c_[0] -= cn * lbn;
129  lbn *= u;
130  }
131  return i;
132  }
133  };
