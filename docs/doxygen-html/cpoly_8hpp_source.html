21  /// Model of a polynomial of a continuous variable.
22  ///
23  /// The general template class cpoly is not fully compatible with \ref
24  /// statdim. The type \a C of each term may be a \ref statdim, in which
25  /// case type \a V must be \ref dyndim. Similarly, if type \a C be \ref
26  /// dyndim, then type \a V must be \ref dyndim.
27  ///
28  /// However, the specialization \ref cpoly<0,V,C> for the zeroth-degree
29  /// polynomial (the constant) is fully compatible with statdim.
30  ///
31  /// Class cpoly inherits from \ref cfunc because cpoly is a model of a
32  /// continuous function.
33  ///
34  /// \tparam D Degree of polynomial.
35  /// \tparam V Type of variable. By default, primitive type double.
36  /// \tparam C Type of each term. By default, primitive type double.
37  template <unsigned D, typename V = double, typename C = double>
38  class cpoly : public cfunc<V, C, cpoly<D, V, C>>
39  {
40  enum {
41  N = D + 1 ///< Number of coefficients.
42  };
43 
44  /// Coefficients of polynomial. Each of these is of the same type as the
45  /// variable. So if the variable be of type dyndim, then so will each
46  /// coefficient be.
47  std::array<V, N> c_;
48 
49  public:
50  /// By default, initialize coefficients to zero.
51  cpoly() { memset(c_.data(), 0, sizeof(c_)); }
52 
53  /// Copy coefficients from an array.
54  cpoly(std::array<V, N> const &cc) : c_(cc) {}
55 
56  /// Copy coefficients from a vector.
57  cpoly(std::vector<V> const &cc)
58  {
59  if (cc.size() != N) {
60  throw "Wrong number of coefficients in vector initializer.";
61  }
62  for (unsigned i = 0; i < N; ++i) {
63  c_[i] = cc[i];
64  }
65  }
66 
67  /// Return reference to mutable coefficients.
68  std::array<V, N> &c() { return c_; }
69 
70  /// Return reference to immutable coefficients.
71  std::array<V, N> const &c() const { return c_; }
72 
73  /// Type of derivative.
74  using DERIV = cpoly<D - 1, V, decltype(C() / V())>;
75 
76  /// Type of integral.
77  using INTEG = cpoly<D + 1, V, decltype(C() * V())>;
78 
79  /// Evaluate polynomial.
80  C operator()(V const &v /**< Independent variable. */) const
81  {
82  C val = c_[0];
83  V vp = v; // Initialize pth power of independent variable.
84  for (unsigned i = 1; i < c_.size(); ++i) {
85  val += c_[i] * vp;
86  vp *= v;
87  }
88  return val;
89  }
90 
91  /// Return function representing derivative.
92  DERIV derivative() const
93  {
94  DERIV d;
95  for (unsigned i = 1; i < c_.size(); ++i) {
96  d.c()[i - 1] = i * c_[i];
97  }
98  return d;
99  }
100 
101  /// Return function representing integral from specified lower bound.
102  INTEG integral(V const &lb /**< Lower bound of integration. */) const
103  {
104  INTEG i;
105  // Initialize constant with zero in right units (in case type C be
106  // dimval).
107  i.c()[0] = 0.0 * (*this)(lb)*lb;
108  V lbn = lb; // Initialize nth power of lower bound.
109  for (unsigned j = 0; j < c_.size(); ++j) {
110  V const cn = c_[j] / (j + 1);
111  i.c()[j + 1] = cn;
112  i.c()[0] -= cn * lbn;
113  lbn *= lb;
114  }
115  return i;
116  }
117  };
