22  /// Model of a polynomial of a continuous variable.
23  ///
24  /// Class cpoly inherits from \ref cfunc because cpoly is a model of a
25  /// continuous function.
26  ///
27  /// \tparam D Degree of polynomial.
28  /// \tparam V Type of variable. By default, primitive type double.
29  /// \tparam C Type of each term. By default, primitive type double.
30  template <unsigned D, typename V = double, typename C = double>
31  class cpoly : public cfunc<V, C, cpoly<D, V, C>>
32  {
33  /// Allow other type of cpoly to access private data. This allows for
34  /// multiplication of polynomials.
35  ///
36  /// Ideally, only another cpoly with the same type of variable should be
37  /// granted friendship, but there is no way to declare a partly
38  /// specialized friend template.
39  ///
40  /// \tparam OD Degree of other polynomial.
41  /// \tparam OV Type of other polynomial's variable.
42  /// \tparam OC Type of other polynomial's term.
43  template <unsigned OD, typename OV, typename OC>
44  friend class cpoly;

47  std::array<C, N> c_; ///< Normalized coefficients.

161  /// Evaluate polynomial.
162  C operator()(/** Value of variable in polynomial. */ V const &v) const
163  {
164  double const vn = v / V(1.0); // Normalized value of variable.
165  double vp = vn; // Initial power of normalized value.
166  C val = c_[0];
167  for (unsigned i = 1; i < c_.size(); ++i) {
168  val += c_[i] * vp;
169  vp *= vn;
170  }
171  return val;
172  }
173 
174  using dterm = decltype(C(1.0) / V(1.0)); ///< Type of derivative's term.
175  using iterm = decltype(C(1.0) * V(1.0)); ///< Type of integral's term.
176  using deriv = cpoly<D - 1, V, dterm>; ///< Type of derivative.
177  using integ = cpoly<D + 1, V, iterm>; ///< Type of integral.
178  friend class cpoly<D - 1, V, dterm>; ///< Integral needs access.
179  friend class cpoly<D + 1, V, iterm>; ///< Derivative needs access.
180 
181  /// Return function representing derivative.
182  deriv derivative() const
183  {
184  deriv d;
185  for (unsigned i = 1; i < c_.size(); ++i) {
186  d.c_[i - 1] = i * c_[i] / V(1.0);
187  }
188  return d;
189  }
190 
191  /// Return function representing integral from specified lower bound.
192  integ integral(/** Lower bound of integration. */ V const &lb) const
193  {
194  integ i;
195  // Initialize constant with zero in right units.
196  i.c_[0] = 0.0 * (*this)(lb)*lb;
197  double const u = lb / V(1.0);
198  double lbn = u; // Initialize power of normalized lower bound.
199  for (unsigned j = 0; j < c_.size(); ++j) {
200  iterm const cn = c_[j] / (j + 1) * V(1.0);
201  i.c_[j + 1] = cn;
202  i.c_[0] -= cn * lbn;
203  lbn *= u;
204  }
205  return i;
206  }
207  };
