21  /// Model of a polynomial of a continuous variable.
22  ///
23  /// This class is not fully compatible with statdim. The type \a C of each
24  /// term may be a statdim, in which case type \a V must be dyndim.
25  /// Similarly, if type \a C be dyndim, then type \a V must be dyndim.
26  ///
27  /// However, the specialization for the zeroth-degree polynomial (the
28  /// constant) is fully compatible with statdim.
29  ///
30  /// Class cpoly inherits from cfunc because cpoly is a model of a continuous
31  /// function.
32  ///
33  /// \tparam D Degree of polynomial.
34  /// \tparam V Type of variable. By default, primitive type double.
35  /// \tparam C Type of each term. By default, primitive type double.
36  template <unsigned D, typename V = double, typename C = double>
37  class cpoly : public cfunc<V, C, cpoly<D, V, C>>
38  {
39  enum {
40  N = D + 1 ///< Number of coefficients.
41  };
42 
43  /// Coefficients of polynomial. Each of these is of the same type as the
44  /// variable. So if the variable be of type dyndim, then so will each
45  /// coefficient be.
46  std::array<V, N> c_;
47 
48  public:
49  /// By default, initialize coefficients to zero.
50  cpoly() { memset(c_.data(), 0, sizeof(c_)); }
51 
52  /// Copy coefficients from an array.
53  cpoly(std::array<V, N> const &cc) : c_(cc) {}
54 
55  /// Copy coefficients from a vector.
56  cpoly(std::vector<V> const &cc)
57  {
58  if (cc.size() != N) {
59  throw "Wrong number of coefficients in vector initializer.";
60  }
61  for (unsigned i = 0; i < N; ++i) {
62  c_[i] = cc[i];
63  }
64  }
65 
66  /// Return reference to mutable coefficients.
67  std::array<V, N> &c() { return c_; }
68 
69  /// Return reference to immutable coefficients.
70  std::array<V, N> const &c() const { return c_; }
71 
72  /// Type of derivative.
73  using DERIV = cpoly<D - 1, V, decltype(C() / V())>;
74 
75  /// Type of integral.
76  using INTEG = cpoly<D + 1, V, decltype(C() * V())>;
77 
78  /// Evaluate polynomial.
79  C operator()(V const &v /**< Independent variable. */) const
80  {
81  C val = c_[0];
82  V vp = v; // Initialize pth power of independent variable.
83  for (unsigned i = 1; i < c_.size(); ++i) {
84  val += c_[i] * vp;
85  vp *= v;
86  }
87  return val;
88  }
89 
90  /// Return function representing derivative.
91  DERIV derivative() const
92  {
93  DERIV d;
94  for (unsigned i = 1; i < c_.size(); ++i) {
95  d.c()[i - 1] = i * c_[i];
96  }
97  return d;
98  }
99 
100  /// Return function representing integral from specified lower bound.
101  INTEG integral(V const &lb /**< Lower bound of integration. */) const
102  {
103  INTEG i;
104  // Initialize constant with zero in right units (in case type C be
105  // dimval).
106  i.c()[0] = 0.0 * (*this)(lb)*lb;
107  V lbn = lb; // Initialize nth power of lower bound.
108  for (unsigned j = 0; j < c_.size(); ++j) {
109  V const cn = c_[j] / (j + 1);
110  i.c()[j + 1] = cn;
111  i.c()[0] -= cn * lbn;
112  lbn *= lb;
113  }
114  return i;
115  }
116  };
