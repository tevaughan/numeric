100  /// Given the value for variable \a y and the value for its derivative \a
101  /// dydx, use the fifth-order Cash-Karp Runge-Kutta method to advance the
102  /// solution for \a y over an interval \a h, and return the incremented
103  /// variable as \a yout. Also, return, by way of the embedded
104  /// fourth-order method, an estimate \a yerr of the local truncation
105  /// error in \a yout. The supplied function \a deriv returns the
106  /// derivative (just like dydx) at any value of the independent variable.
107  ///
108  /// This function is based on a simplified version of `rkck()` found on
109  /// Page 719 in Numerical Recipes in C, Second Edition. The
110  /// simplification is due mainly to the fact that the present function is
111  /// used for quadrature, and so \a deriv does not require \a y as input.
112  ///
113  void rkck(X const &h, ///< Length of interval.
114  Y &yout, ///< Accumulated value at end of interval.
115  Y &yerr ///< Estimate of local truncation error.
116  )
117  {
118  // 1st step is given as dydx on input.
119  // 2nd step not needed because deriv() does not need y as input.
120  X const x3 = x + a3 * h;
121  X const x4 = x + a4 * h;
122  X const x5 = x + a5 * h;
123  X const x6 = x + a6 * h;
124  RAT<Y, X> const ak3 = deriv(x3); // 3rd step.
125  RAT<Y, X> const ak4 = deriv(x4); // 4th step.
126  RAT<Y, X> const ak5 = deriv(x5); // 5th step.
127  RAT<Y, X> const ak6 = deriv(x6); // 6th step.
128  if (store_dydx) {
129  fi_list.push_back({x3, ak3});
130  fi_list.push_back({x4, ak4});
131  fi_list.push_back({x5, ak5});
132  fi_list.push_back({x6, ak6});
133  }
134  // Accumulate increments with proper weights.
135  yout = y + h * (c1 * dydx + c3 * ak3 + c4 * ak4 + c6 * ak6);
136  // Estimate error as difference between fourth- and fifth-order
137  // methods.
138  yerr = h *
139  (dc1 * dydx + dc3 * ak3 + dc4 * ak4 + dc5 * ak5 + dc6 * ak6);
140  }

142  /// Fifth-order Runge-Kutta step with monitoring of local truncation
143  /// error to ensure accurcy and adjust stepsize. Input are the
144  /// accumulated variable \a y and its derivative \a dydx at the starting
145  /// value of the independent variable \a x. Also input are the stepsize
146  /// \a htry to be attempted and the required accuracy \a tol. On output,
147  /// \a y and \a x are replaced by their new values, \a hdid is the
148  /// stepsize that was actually accomplished, and \a hnext is the
149  /// estimated next stepsize. The supplied function \a deriv returns the
150  /// derivative (just like dydx) at any value of the independent variable.
151  ///
152  /// This function is based on `rkqs()` found on Page 719 in Numerical
153  /// Recipes in C, Second Edition.
154  ///
155  void rkqs(X const &htry, ///< Stepsize to be attempted.
156  Y const &yscal, ///< Scaling used to monitor accuracy.
157  X &hdid, ///< Stepsize that was accomplished.
158  X &hnext ///< Estimated next stepsize.
159  )
160  {
161  static double constexpr SAFETY = 0.9, PGROW = -0.2, PSHRNK = -0.25;
162  static double const ERRCON = pow(5.0 / SAFETY, 1.0 / PGROW);
163  double err;
164  Y yerr;
165  Y ytemp;
166  X h = htry; // Set stepsize to the initial trial value.
167  while (true) {
168  rkck(h, ytemp, yerr); // Take a step.
169  err = fabs(yerr / yscal / tol);
170  if (err <= 1.0) {
171  break;
172  }
173  X const htemp = SAFETY * h * pow(err, PSHRNK);
174  // Truncation error is too large. Reduce stepsize no more than a
175  // factor of 10.
176  X const tenth = 0.1 * h;
177  static X const zero(0.0);
178  if (h >= zero) {
179  if (htemp > tenth) {
180  h = htemp;
181  } else {
182  h = tenth;
183  }
184  } else {
185  if (htemp < tenth) {
186  h = htemp;
187  } else {
188  h = tenth;
189  }
190  }
191  X const xnew = x + h;
192  if (xnew == x) {
193  std::cerr << "rkqs: WARNING: stepsize underflow" << std::endl;
194  h = std::numeric_limits<double>::epsilon() * 100.0 * x;
195  break;
196  }
197  }
198  // Increase stepsize no more than a factor of 5.
199  if (err > ERRCON) {
200  hnext = SAFETY * h * pow(err, PGROW);
201  } else {
202  hnext = 5.0 * h;
203  }
204  x += (hdid = h);
205  y = ytemp;
206  }

208  /// This function is based on `odeint()` on Page 721 of Numerical Recipes
209  /// in C, Second Edition.
210  ///
211  void init(X x1, ///< Lower limit of integration.
212  X x2, ///< Upper limit of integration.
213  int n ///< Inverse of initial step size.
214  )
215  {
216  double constexpr eps = std::numeric_limits<double>::epsilon();
217  double constexpr min_tol = 100.0 * eps;
218  if (tol <= 0.0) {
219  throw "tolerance not positive";
220  } else if (tol < min_tol) {
221  tol = min_tol;
222  }
223  X const hmin(0.0);
224  if (n < 2) {
225  n = 2;
226  }
227  X const h1 = (X(x2) - X(x1)) / (n - 1);
228  X h;
229  if (x2 - x1 > X(0.0)) {
230  h = +fabs(h1);
231  } else {
232  h = -fabs(h1);
233  }
234  ilist<X, Y> ii_list;
235  while (true) {
236  dydx = deriv(x);
237  static Y const TINY(1.0E-300);
238  // General-purpose scaling used to monitor accuracy.
239  Y const yscal = fabs(y) + fabs(dydx * h) + TINY;
240  if (store_dydx) {
241  fi_list.push_back({x, dydx});
242  }
243  if (store_y) {
244  ii_list.push_back({x, y});
245  }
246  X const xh = x + h;
247  using XSQR = PRD<X, X>;
248  if ((xh - x2) * (xh - x1) > XSQR(0.0)) {
249  h = x2 - x; // Decrease stepsize to avoid overshoot.
250  }
251  X hdid, hnext;
252  rkqs(h, yscal, hdid, hnext);
253  if (hdid == h) {
254  ++nok;
255  } else {
256  ++nbad;
257  }
258  if ((x - x2) * (x2 - x1) >= XSQR(0.0)) {
259  if (store_dydx) {
260  fi = interpolant<X, DYDX>(fi_list);
261  }
262  if (store_y) {
263  ii_list.push_back({x, y});
264  ii = interpolant<X, Y>(ii_list);
265  }
266  return; // We are done; exit normally.
267  }
268  if (fabs(hnext) <= hmin) {
269  std::cerr << "rk_quad: WARNING: step size too small"
270  << std::endl;
271  return;
272  }
273  h = hnext;
274  }
275  std::cerr << "rk_quad: WARNING: too many steps" << std::endl;
276  }

367  /// Numerically integrate a function, and return the result.
368  ///
369  /// Use the trapezoidal rule with adaptive quadrature.
370  ///
371  /// The specified tolerance is on the relative accuracy of the estimated
372  /// mean value of the function over each subdivided interval of the variable
373  /// of integration. Every interval is recursively subdivided into equal
374  /// halves. At each stage in the subdivision, the estimated mean height of
375  /// the function over the undivided interval is compared to the refined
376  /// estimate after subdivision of the interval. If the magnitude of the
377  /// difference between these estimates be smaller than the product of the
378  /// tolerance and the refined estimate, then recursion at the current
379  /// interval ceases.
380  ///
381  /// Recursion is not done by way of the function-call stack but rather by
382  /// way of a stack with dynamically allocated memory.
383  ///
384  /// The basic idea for the algorithm is found in the notes of Jim Lambers.
385  /// http://www.math.usm.edu/lambers/mat460/fall09/lecture30.pdf
386  ///
387  /// The present, original C++-11 implementation is due entirely to Thomas E.
388  /// Vaughan, who wrote it in 2016. The meaning of the tolerance is
389  /// different in Vaughan's implementation, which also adds a test for the
390  /// numerical limitation of double-precision numbers.
391  ///
392  /// \tparam R Type of instance returned by function.
393  /// \tparam A Type of argument to function.
394  /// \tparam A1 Type of lower limit of integration; A1 must convert to A.
395  /// \tparam A2 Type of upper limit of integration; A2 must convert to A.
396  /// \return Numeric integral of function.
397  template <typename R, typename A, typename A1, typename A2>
398  PRD<R, A>
399  integral(std::function<R(A)> f, ///< Function to be integrated.
400  A1 aa, ///< Lower limit of integration.
401  A2 bb, ///< Upper limit of integration.
402  double t = 1.0E-06, ///< Error tolerance.
403  unsigned n = 16 ///< Initial number of evenly spaced samples.
404  )
405  {
406  double constexpr eps = std::numeric_limits<double>::epsilon();
407  double constexpr min_tol = 1000.0 * eps;
408  double tol = t;
409  if (tol <= 0.0) {
410  throw "tolerance not positive";
411  } else if (tol < min_tol) {
412  tol = min_tol;
413  }
414  double sign = 1.0;
415  A a = aa;
416  A b = bb;
417  if (a > b) {
418  std::swap(a, b);
419  sign = -1.0;
420  }
421  subinterval_stack<A, R> s(n, a, b, f);
422  using I = PRD<R, A>;
423  integral_stats<I> stats(0.0 * aa * f(aa));
424  while (s.size()) {
425  using interval = interval<A, R>;
426  interval const r = *s.rbegin();
427  s.pop_back();
428  A const midp = 0.5 * (r.a + r.b); // midpoint of interval
429  R const fmid = f(midp); // function value at midpoint
430  A const len = r.b - r.a; // length of interval
431  R const mean = 0.5 * (r.fa + r.fb); // mean of function values
432  R const rmean = 0.5 * (mean + fmid); // refined mean
433  // Trapezoidal rule len*mean is estimate for current subinterval.
434  // Composite trapezoidal rule 0.5*len*(mean + fmid) is refined
435  // estimate obtained by subdividing the current subinterval.
436  // Magnitude of difference between them is u1. Stop refining when u1
437  // become smaller than tolerance t times magnitude of refined
438  // estimate.
439  R const u0 = fabs(rmean);
440  R const u1 = fabs(mean - rmean);
441  R const u2 = fabs(mean + rmean);
442  R const u3 = u0 * tol;
443  I const ds = rmean * len;
444  if (u1 <= u3) {
445  // Estimated error sufficiently small. Stop refining estimate.
446  stats.add(ds, u1 * len);
447  } else if (u1 <= u2 * tol) {
448  // Calculated error too small. Stop refining estimate.
449  stats.add(ds, u1 * len);
450  } else if (len <= fabs(midp) * tol) {
451  // Length of interval too small. Stop refining estimate.
452  stats.add(ds, u1 * len);
453  } else if (fabs(ds) <= fabs(stats.area()) * tol) {
454  // Increment to integral too small. Stop refining estimate.
455  stats.add(ds, u1 * len);
456  } else {
457  // Continue refining estimate.
458  s.push_back(interval{r.a, midp, r.fa, fmid});
459  s.push_back(interval{midp, r.b, fmid, r.fb});
460  }
461  }
462  I const farea = fabs(stats.area());
463  I const sigma = stats.stdev(); // statistical error
464  I const derr = farea * t; // desired error
465  I const rerr = farea * eps; // round-off error
466  I eerr; // greater of statistical and round-off error
467  if (sigma < rerr) {
468  eerr = rerr;
469  } else {
470  eerr = sigma;
471  }
472  if (eerr > derr) {
473  std::cerr << "integral: WARNING: Estimated error " << eerr / farea
474  << " is greater than tolerance " << t << "." << std::endl;
475  }
476  return sign * stats.area();
477  }
