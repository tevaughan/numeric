26  /// Numerically integrate a function, and return the result.
27  ///
28  /// Use the trapezoidal rule with adaptive quadrature.
29  ///
30  /// The specified tolerance is on the relative accuracy of the estimated
31  /// mean value of the function over each subdivided interval of the variable
32  /// of integration. Every interval is recursively subdivided into equal
33  /// halves. At each stage in the subdivision, the estimated mean height of
34  /// the function over the undivided interval is compared to the refined
35  /// estimate after subdivision of the interval. If the magnitude of the
36  /// difference between these estimates be smaller than the product of the
37  /// tolerance and the refined estimate, then recursion at the current
38  /// interval ceases.
39  ///
40  /// Recursion is not done by way of the function-call stack but rather by
41  /// way of a stack with dynamically allocated memory.
42  ///
43  /// The basic idea for the algorithm is found in the notes of Jim Lambers.
44  /// http://www.math.usm.edu/lambers/mat460/fall09/lecture30.pdf
45  ///
46  /// The present, original C++-11 implementation is due entirely to Thomas E.
47  /// Vaughan, who wrote it in 2016. The meaning of the tolerance is
48  /// different in Vaughan's implementation, which also adds a test for the
49  /// numerical limitation of double-precision numbers.
50  ///
51  /// \tparam R Type of instance returned by function.
52  /// \tparam A Type of argument to function.
53  /// \tparam A1 Type of lower limit of integration; A1 must convert to A.
54  /// \tparam A2 Type of upper limit of integration; A2 must convert to A.
55  /// \param f Function to be integrated.
56  /// \param aa Lower limit of integration.
57  /// \param bb Upper limit of integration.
58  /// \param t Error tolerance.
59  /// \param n Initial number of evenly spaced samples of the function.
60  /// \return Numeric integral of function.
61  template <typename R, typename A, typename A1, typename A2>
62  auto integral(std::function<R(A)> f, A1 aa, A2 bb, double t = 1.0E-06,
63  unsigned n = 16)
64  // Meyers, Scott. Effective Modern C++ (first ed., 2015), page 28.
65  -> decltype(std::forward<std::function<R(A)>>(f)(A()) * A())
66  {
67  double constexpr eps = std::numeric_limits<double>::epsilon();
68  double constexpr min_tol = 1000.0 * eps;
69  double tol = t;
70  if (tol <= 0.0) {
71  throw "tolerance not positive";
72  } else if (tol < min_tol) {
73  tol = min_tol;
74  }
75  double sign = 1.0;
76  A a = aa;
77  A b = bb;
78  if (a > b) {
79  std::swap(a, b);
80  sign = -1.0;
81  }
82  subinterval_stack<A, R> s(n, a, b, f);
83  using I = decltype(std::forward<std::function<R(A)>>(f)(A()) * A());
84  integral_stats<I> stats;
85  while (s.size()) {
86  using interval = interval<A, R>;
87  interval const r = *s.rbegin();
88  s.pop_back();
89  A const midp = 0.5 * (r.a + r.b); // midpoint of interval
90  R const fmid = f(midp); // function value at midpoint
91  A const len = r.b - r.a; // length of interval
92  R const mean = 0.5 * (r.fa + r.fb); // mean of function values
93  R const rmean = 0.5 * (mean + fmid); // refined mean
94  // Trapezoidal rule len*mean is estimate for current subinterval.
95  // Composite trapezoidal rule 0.5*len*(mean + fmid) is refined
96  // estimate obtained by subdividing the current subinterval.
97  // Magnitude of difference between them is u1. Stop refining when u1
98  // become smaller than tolerance t times magnitude of refined
99  // estimate.
100  R const u0 = fabs(rmean);
101  R const u1 = fabs(mean - rmean);
102  R const u2 = fabs(mean + rmean);
103  R const u3 = u0 * tol;
104  I const ds = rmean * len;
105  if (u1 <= u3) {
106  // Estimated error sufficiently small. Stop refining estimate.
107  stats.add(ds, u1 * len);
108  } else if (u1 <= u2 * tol) {
109  // Calculated error too small. Stop refining estimate.
110  stats.add(ds, u1 * len);
111  } else if (len <= fabs(midp) * tol) {
112  // Length of interval too small. Stop refining estimate.
113  stats.add(ds, u1 * len);
114  } else if (fabs(ds) <= fabs(stats.area()) * tol) {
115  // Increment to integral too small. Stop refining estimate.
116  stats.add(ds, u1 * len);
117  } else {
118  // Continue refining estimate.
119  s.push_back(interval{r.a, midp, r.fa, fmid});
120  s.push_back(interval{midp, r.b, fmid, r.fb});
121  }
122  }
123  I const farea = fabs(stats.area());
124  I const sigma = stats.stdev(); // statistical error
125  I const nres = I(1.0) + farea; // normalized result
126  I const derr = nres * t; // desired error
127  I const rerr = nres * eps; // round-off error
128  I eerr; // greater of statistical and round-off error
129  if (sigma < rerr) {
130  eerr = rerr;
131  } else {
132  eerr = sigma;
133  }
134  if (eerr > derr) {
135  std::cerr << "integral: WARNING: Estimated error " << eerr / nres
136  << " is greater than tolerance " << t << "." << std::endl;
137  }
138  return sign * stats.area();
139  }
