27  /// Given the value for variable \a y and the value for its derivative \a
28  /// dydx, use the fifth-order Cash-Karp Runge-Kutta method to advance the
29  /// solution for \a y over an interval \a h, and return the incremented
30  /// variable as \a yout. Also, return, by way of the embedded fourth-order
31  /// method, an estimate \a yerr of the local truncation error in \a yout.
32  /// The supplied function \a deriv returns the derivative (just like dydx)
33  /// at any value of the independent variable.
34  ///
35  /// This function is based on a simplified version of `rkck()` found on Page
36  /// 719 in Numerical Recipes in C, Second Edition. The simplification is
37  /// due mainly to the fact that the present function is used for quadrature,
38  /// and so \a deriv does not require \a y as input.
39  ///
40  /// \tparam X Type of independent variable.
41  /// \tparam Y Type of variable that is accumulated during integration.
42  template <typename X, typename Y>
43  void rkck(Y const &y, ///< Initial value of accumulated variable.
44  RAT<Y, X> const &dydx, ///< Derivative at beginning of interval.
45  X const &x, ///< Initial value of independent variable.
46  X const &h, ///< Length of interval.
47  Y &yout, ///< Accumulated value at end of interval.
48  Y &yerr, ///< Estimate of local truncation error.
49  std::function<RAT<Y, X>(X)> deriv ///< Function for derivative.
50  )
51  {
52  double constexpr a3 = 0.3, a4 = 0.6, a5 = 1.0, a6 = 0.875;
53  double constexpr c1 = 37.0 / 378.0, c3 = 250.0 / 621.0;
54  double constexpr c4 = 125.0 / 594.0, c6 = 512.0 / 1771.0;
55  double constexpr dc1 = c1 - 2825.0 / 27648.0;
56  double constexpr dc3 = c3 - 18575.0 / 48384.0;
57  double constexpr dc4 = c4 - 13525.0 / 55296.0;
58  double constexpr dc5 = -277.0 / 14336.0;
59  double constexpr dc6 = c6 - 0.25;
60  // 1st step is given as dydx on input.
61  // 2nd step not needed because deriv() does not need y as input.
62  RAT<Y, X> const ak3 = deriv(x + a3 * h); // 3rd step.
63  RAT<Y, X> const ak4 = deriv(x + a4 * h); // 4th step.
64  RAT<Y, X> const ak5 = deriv(x + a5 * h); // 5th step.
65  RAT<Y, X> const ak6 = deriv(x + a6 * h); // 6th step.
66  // Accumulate increments with proper weights.
67  yout = y + h * (c1 * dydx + c3 * ak3 + c4 * ak4 + c6 * ak6);
68  // Estimate error as difference between fourth- and fifth-order methods.
69  yerr = h * (dc1 * dydx + dc3 * ak3 + dc4 * ak4 + dc5 * ak5 + dc6 * ak6);
70  }

72  /// Fifth-order Runge-Kutta step with monitoring of local truncation error
73  /// to ensure accurcy and adjust stepsize. Input are the accumulated
74  /// variable \a y and its derivative \a dydx at the starting value of the
75  /// independent variable \a x. Also input are the stepsize \a htry to be
76  /// attempted and the required accuracy \a eps. On output, \a y and \a x
77  /// are replaced by their new values, \a hdid is the stepsize that was
78  /// actually accomplished, and \a hnext is the estimated next stepsize. The
79  /// supplied function \a deriv returns the derivative (just like dydx) at
80  /// any value of the independent variable.
81  ///
82  /// This function is based on `rkqs()` found on Page 719 in Numerical
83  /// Recipes in C, Second Edition.
84  template <typename X, typename Y>
85  void rkqs(Y &y, ///< Accumulated variable.
86  RAT<Y, X> const &dydx, ///< Derivative at beginning of interval.
87  X &x, ///< Independent variable.
88  X const &htry, ///< Stepsize to be attempted.
89  double eps, ///< Required accuracy.
90  Y const &yscal, ///< Scaling used to monitor accuracy.
91  X &hdid, ///< Stepsize that was accomplished.
92  X &hnext, ///< Estimated next stepsize.
93  std::function<RAT<Y, X>(X)> deriv ///< Function for derivative.
94  )
95  {
96  double constexpr SAFETY = 0.9, PGROW = -0.2, PSHRNK = -0.25;
97  double constexpr ERRCON = pow(5.0 / SAFETY, 1.0 / PGROW);
98  double err;
99  Y yerr;
100  Y ytemp;
101  X h = htry; // Set stepsize to the initial trial value.
102  while (true) {
103  rkck(y, dydx, x, h, ytemp, yerr, deriv); // Take a step.
104  err = fabs(yerr / yscal / eps);
105  if (err <= 1.0) {
106  break;
107  }
108  X const htemp = SAFETY * h * pow(err, PSHRNK);
109  // Truncation error is too large. Reduce stepsize no more than a
110  // factor of 10.
111  X const tenth = 0.1 * h;
112  static X const zero(0.0);
113  if (h >= zero) {
114  if (htemp > tenth) {
115  h = htemp;
116  } else {
117  h = tenth;
118  }
119  } else {
120  if (htemp < tenth) {
121  h = htemp;
122  } else {
123  h = tenth;
124  }
125  }
126  X const xnew = x + h;
127  if (xnew == x) {
128  std::cerr << "rkqs: WARNING: stepsize underflow" << std::endl;
129  break;
130  }
131  }
132  // Increase stepsize no more than a factor of 5.
133  if (err > ERRCON) {
134  hnext = SAFETY * h * pow(err, PGROW);
135  } else {
136  hnext = 5.0 * h;
137  }
138  x += (hdid = h);
139  y = ytemp;
140  }

148  /// Numerically integrate a function, and return the result. Use
149  /// fifth-order Runge-Kutta with adaptive stepsize for quadrature. The
150  /// initial guess for the step size is used at the lower limit of
151  /// integration.
152  ///
153  /// integral_rk() optionally returns by reference an approximant for the
154  /// function integrated and an approximant for the indefinite integral,
155  /// zeroed at the lower limit of integration.
156  ///
157  /// This function is based on `odeint()` on Page 721 of Numerical Recipes in
158  /// C, Second Edition.
159  ///
160  /// \tparam R Type of instance returned by function.
161  /// \tparam A Type of argument to function.
162  /// \tparam A1 Type of lower limit of integration; A1 must convert to A.
163  /// \tparam A2 Type of upper limit of integration; A2 must convert to A.
164  /// \return Numeric integral of function.
165  template <typename R, typename A, typename A1, typename A2>
166  PRD<R, A>
167  integral_rk(std::function<R(A)> f, ///< Function to be integrated.
168  A1 x1, ///< Lower limit of integration.
169  A2 x2, ///< Upper limit of integration.
170  A h1, ///< Initial guess for step size.
171  double t = 1.0E-06, ///< Error tolerance.
172  /// If non-null, output approximant for function \a f.
173  interpolant<A, R> *fi = nullptr,
174  /// If non-null, output approximant for indef. integral.
175  interpolant<A, PRD<R, A>> *ii = nullptr)
176  {
177  using Y = PRD<R, A>;
178  A const hmin(0.0);
179  A x = x1;
180  A h;
181  if (x2 - x1 > A(0.0)) {
182  h = +fabs(h1);
183  } else {
184  h = -fabs(h1);
185  }
186  int nok = 0, nbad = 0;
187  Y y(0.0);
188  int constexpr MAXSTP = 10000;
189  ilist<A, R> fi_list;
190  ilist<A, Y> ii_list;
191  for (int nstp = 0; nstp < MAXSTP; ++nstp) {
192  R const dydx = f(x);
193  static Y const TINY(1.0E-300);
194  // General-purpose scaling used to monitor accuracy.
195  Y const yscal = fabs(y) + fabs(dydx * h) + TINY;
196  if (fi) {
197  fi_list.push_back({x, dydx});
198  }
199  if (ii) {
200  ii_list.push_back({x, y});
201  }
202  A const xh = x + h;
203  using X2 = PRD<A, A>;
204  if ((xh - x2) * (xh - x1) > X2(0.0)) {
205  h = x2 - x; // Decrease stepsize to avoid overshoot.
206  }
207  A hdid, hnext;
208  rkqs(y, dydx, x, h, t, yscal, hdid, hnext, f);
209  if (hdid == h) {
210  ++nok;
211  } else {
212  ++nbad;
213  }
214  if ((x - x2) * (x2 - x1) >= X2(0.0)) {
215  if (fi) {
216  *fi = interpolant<A, R>(fi_list);
217  }
218  if (ii) {
219  *ii = interpolant<A, Y>(ii_list);
220  }
221  return y; // We are done; exit normally.
222  }
223  if (fabs(hnext) <= hmin) {
224  std::cerr << "integral_rk: WARNING: step size too small"
225  << std::endl;
226  return y;
227  }
228  h = hnext;
229  }
230  std::cerr << "integral_rk: WARNING: too many steps" << std::endl;
231  return y;
232  }

258  /// Numerically integrate a function, and return the result.
259  ///
260  /// Use the trapezoidal rule with adaptive quadrature.
261  ///
262  /// The specified tolerance is on the relative accuracy of the estimated
263  /// mean value of the function over each subdivided interval of the variable
264  /// of integration. Every interval is recursively subdivided into equal
265  /// halves. At each stage in the subdivision, the estimated mean height of
266  /// the function over the undivided interval is compared to the refined
267  /// estimate after subdivision of the interval. If the magnitude of the
268  /// difference between these estimates be smaller than the product of the
269  /// tolerance and the refined estimate, then recursion at the current
270  /// interval ceases.
271  ///
272  /// Recursion is not done by way of the function-call stack but rather by
273  /// way of a stack with dynamically allocated memory.
274  ///
275  /// The basic idea for the algorithm is found in the notes of Jim Lambers.
276  /// http://www.math.usm.edu/lambers/mat460/fall09/lecture30.pdf
277  ///
278  /// The present, original C++-11 implementation is due entirely to Thomas E.
279  /// Vaughan, who wrote it in 2016. The meaning of the tolerance is
280  /// different in Vaughan's implementation, which also adds a test for the
281  /// numerical limitation of double-precision numbers.
282  ///
283  /// \tparam R Type of instance returned by function.
284  /// \tparam A Type of argument to function.
285  /// \tparam A1 Type of lower limit of integration; A1 must convert to A.
286  /// \tparam A2 Type of upper limit of integration; A2 must convert to A.
287  /// \return Numeric integral of function.
288  template <typename R, typename A, typename A1, typename A2>
289  PRD<R, A>
290  integral(std::function<R(A)> f, ///< Function to be integrated.
291  A1 aa, ///< Lower limit of integration.
292  A2 bb, ///< Upper limit of integration.
293  double t = 1.0E-06, ///< Error tolerance.
294  unsigned n = 16 ///< Initial number of evenly spaced samples.
295  )
296  {
297  double constexpr eps = std::numeric_limits<double>::epsilon();
298  double constexpr min_tol = 1000.0 * eps;
299  double tol = t;
300  if (tol <= 0.0) {
301  throw "tolerance not positive";
302  } else if (tol < min_tol) {
303  tol = min_tol;
304  }
305  double sign = 1.0;
306  A a = aa;
307  A b = bb;
308  if (a > b) {
309  std::swap(a, b);
310  sign = -1.0;
311  }
312  subinterval_stack<A, R> s(n, a, b, f);
313  using I = PRD<R, A>;
314  integral_stats<I> stats;
315  while (s.size()) {
316  using interval = interval<A, R>;
317  interval const r = *s.rbegin();
318  s.pop_back();
319  A const midp = 0.5 * (r.a + r.b); // midpoint of interval
320  R const fmid = f(midp); // function value at midpoint
321  A const len = r.b - r.a; // length of interval
322  R const mean = 0.5 * (r.fa + r.fb); // mean of function values
323  R const rmean = 0.5 * (mean + fmid); // refined mean
324  // Trapezoidal rule len*mean is estimate for current subinterval.
325  // Composite trapezoidal rule 0.5*len*(mean + fmid) is refined
326  // estimate obtained by subdividing the current subinterval.
327  // Magnitude of difference between them is u1. Stop refining when u1
328  // become smaller than tolerance t times magnitude of refined
329  // estimate.
330  R const u0 = fabs(rmean);
331  R const u1 = fabs(mean - rmean);
332  R const u2 = fabs(mean + rmean);
333  R const u3 = u0 * tol;
334  I const ds = rmean * len;
335  if (u1 <= u3) {
336  // Estimated error sufficiently small. Stop refining estimate.
337  stats.add(ds, u1 * len);
338  } else if (u1 <= u2 * tol) {
339  // Calculated error too small. Stop refining estimate.
340  stats.add(ds, u1 * len);
341  } else if (len <= fabs(midp) * tol) {
342  // Length of interval too small. Stop refining estimate.
343  stats.add(ds, u1 * len);
344  } else if (fabs(ds) <= fabs(stats.area()) * tol) {
345  // Increment to integral too small. Stop refining estimate.
346  stats.add(ds, u1 * len);
347  } else {
348  // Continue refining estimate.
349  s.push_back(interval{r.a, midp, r.fa, fmid});
350  s.push_back(interval{midp, r.b, fmid, r.fb});
351  }
352  }
353  I const farea = fabs(stats.area());
354  I const sigma = stats.stdev(); // statistical error
355  I const nres = I(1.0) + farea; // normalized result
356  I const derr = nres * t; // desired error
357  I const rerr = nres * eps; // round-off error
358  I eerr; // greater of statistical and round-off error
359  if (sigma < rerr) {
360  eerr = rerr;
361  } else {
362  eerr = sigma;
363  }
364  if (eerr > derr) {
365  std::cerr << "integral: WARNING: Estimated error " << eerr / nres
366  << " is greater than tolerance " << t << "." << std::endl;
367  }
368  return sign * stats.area();
369  }
