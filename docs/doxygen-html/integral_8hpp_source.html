19  /// Use fifth-order Runge-Kutta with adaptive stepsize. See
20  /// rk_quad::rk_quad().
21  ///
22  /// The initial-guess parameter is the inverse fraction of the domain
23  /// (between limits of integration) to use as the guess for the initial step
24  /// size.
25  ///
26  /// \tparam X Type of argument to function.
27  /// \tparam X1 Type of lower limit of integration (convertible to X).
28  /// \tparam X2 Type of upper limit of integration (convertible to X).
29  /// \tparam Y Type returned by function that is to be integrated.
30  /// \return Numeric integral of function.
31  template <typename X, typename Y, typename X1, typename X2>
32  PRD<X, Y> integral(

34  /** Lower limit of integration. */ X1 aa,

36  /** Error tolerance. */ double t = 1.0E-06,
