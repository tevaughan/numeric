12  /// A [piecewise function](https://en.wikipedia.org/wiki/Piecewise) that, in
13  /// logarithmic time, looks up the appropriate sub-function.
14  ///
15  /// The argument \f$ a \f$ to the function is of type \a A.
16  ///
17  /// For any \f$ a \f$, what is found in logarithmic time are the
18  /// corresponding sub-function and its sub-domain. The corresponding
19  /// sub-function, stored in a table, is \f$ f_i \f$ of type \a F, and the
20  /// center of the corresponding subdomain is \f$ a_i \f$, where \f$ i \in
21  /// \{0, 1, \ldots, n-1\} \f$. The sub-argument \f$ a - a_i \f$ is passed
22  /// to \f$ f_i \f$, and the value returned by \f$ f_i \f$ is returned from
23  /// the lookup.
24  ///
25  /// In the simplest, fastest table, each \f$ f_i \f$ ignores its argument
26  /// and returns a constant value. However, the infrastructure provided by
27  /// sparse_table allows for logarithmic-time lookup of any kind of
28  /// sub-function (for example, a cubic interpolant).
29  ///
30  /// To enable logarithmic-time lookup, an instance of sparse_table stores a
31  /// table of \f$ n \f$ triplets \f$ (a_0, {\Delta a}_0, f_0), (a_1, {\Delta
32  /// a}_1, f_1), \ldots, (a_{n-1}, {\Delta a}_{n-1}, f_{n-1}) \f$, sorted so
33  /// that \f$ a_0 < a_1 < \cdots < a_{n-1} \f$.
34  ///
35  /// The lookup occurs by way of operator()(). When \f$ a > a_0 \f$ and \f$
36  /// a < a_{n-1} \f$, record \f$ i \f$ is found so that
37  /// \f[
38  /// -\frac{{\Delta a}_i}{2} < a - a_i \leq +\frac{{\Delta a}_i}{2}.
39  /// \f]
40  /// What operator()() returns is \f$ f_i(a - a_i) \f$.
41  ///
42  /// See dense_table for a type that usually can approximate a given function
43  /// so precisely as sparse_table but with faster lookup. The cost is that
44  /// dense_table requires more storage for the same precision with the same
45  /// sub-function type \a F.
46  ///
47  /// \tparam A Type of function's argument.
48  /// \tparam F Type of each sub-function \f$ f_i \f$ in the table.
49  template <typename A, typename F>
50  class sparse_table
51  {
52  using I = decltype(1.0 / A());
53  A a_frst_; ///< First tabulated argument.
54  A da_; ///< Common difference between subsequent tabulated arguments.
55  I ida_; ///< Inverse of difference between subsequent tabulated args.
56  std::vector<F> f_; ///< Table of sub-functions.
57

61  A const &delta, ///< Common difference \f$ \Delta a \f$.
62  /// Sub-function objects \f$ f_0, \ldots, f_{n-1} \f$.
63  std::vector<F> vf)
64  : a_frst_(first), da_(delta), ida_(1.0 / da_), f_(vf)
65  {
66  }

81  /// For \f$ f_i \f$ and \f$ a_i \f$ associated with an argument \f$ a \f$
82  /// to the function, calculate \f$ f_i(a - a_i) \f$.
83  ///
84  /// - If \f$ a < a_0 \f$, then choose \f$ i = 0 \f$.
85  ///
86  /// - If \f$ a > a_{n-1} \f$, then choose \f$ i = n - 1 \f$.
87  ///
88  /// - Otherwise, choose \f$ i \f$ to be the integer nearest to \f$
89  /// \frac{a - a_0}{\Delta a} + 0.5 \f$.
90  ///
91  /// \return \f$ f_i(a - a_i) \f$.
92  auto operator()(A const &a /**< Argument to function. */) const
93  -> decltype(F()(a))
94  {
95  int i;
96  A ai;
97  if (a < a_frst_) {
98  i = 0;
99  ai = a_frst_;
100  } else {
101  A const last = a_last();
102  if (a > last) {
103  i = f_.size() - 1;
104  ai = last;
105  } else {
106  i = int((a - a_frst_) * ida_ + 0.5);
107  ai = a_frst_ + i * da_;
108  }
109  }
110  return f_[i](a - ai);
111  }
112  };
