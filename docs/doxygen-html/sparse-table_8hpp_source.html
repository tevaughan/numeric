18  /// A [piecewise function](https://en.wikipedia.org/wiki/Piecewise) that, in
19  /// logarithmic time, looks up the sub-function appropriate to the argument.
20  ///
21  /// For any argument \f$ a \f$, the corresponding sub-function and its
22  /// sub-domain are found in logarithmic time. The sub-domains are
23  /// contiguous and of variable length. If \f$ a \f$ be less than the least
24  /// element of the first subdomain or greater than the greatest element of
25  /// the last subdomain, then there is no corresponding sub-function, and
26  /// zero is returned.
27  ///
28  /// Stored in a table of \f$ n \f$ records, the sub-function corresponding
29  /// to \f$ a \f$ is \f$ f_i \f$, and the center of the corresponding
30  /// subdomain is \f$ a_i \f$, where \f$ i \in \{0, 1, \ldots, n-1\} \f$.
31  /// The sub-argument \f$ a - a_i \f$ is passed to \f$ f_i \f$, and the value
32  /// returned by \f$ f_i \f$ is returned from the lookup.
33  ///
34  /// In the simplest, fastest table, each \f$ f_i \f$ ignores its argument
35  /// and returns a constant value. However, the infrastructure provided by
36  /// sparse_table allows for logarithmic-time lookup of any kind of
37  /// sub-function (for example, a cubic interpolant).
38  ///
39  /// To implement logarithmic-time lookup, an instance of sparse_table stores
40  /// \f$ n \f$ triplets \f$ (a_0, \Delta a_0, f_0) \f$, \f$ (a_1, \Delta a_1,
41  /// f_1) \f$, \f$ \ldots \f$, \f$ (a_{n-1}, \Delta a_{n-1}, f_{n-1}) \f$,
42  /// sorted so that \f$ a_0 < a_1 < \cdots < a_{n-1} \f$, where \f$ \Delta
43  /// a_i \f$ is the length of the sub-domain whose center is \f$ a_i \f$.
44  ///
45  /// The lookup occurs by way of operator()(). When \f$ a > a_0 -
46  /// \frac{\Delta a_0}{2} \f$, and \f$ a < a_{n-1} + \frac{\Delta a_{n-1}}{2}
47  /// \f$, record \f$ i \f$ is found so that \f$ |a - a_i| \leq \frac{{\Delta
48  /// a}_i}{2} \f$.
49  ///
50  /// operator()() returns \f$ f_i(a - a_i) \f$.
51  ///
52  /// See dense_table for a type that usually can approximate a given function
53  /// so precisely as sparse_table but with faster lookup. The cost is that
54  /// dense_table requires more storage for the same precision with the same
55  /// sub-function type \a F.
56  ///
57  /// \tparam A Type of sub-function's argument.
58  /// \tparam F Type of sub-function.
59  template <typename A, typename F>
60  class sparse_table
61  {
62  public:
63  /// Type of record in table.
64  struct rec {
65  A a; ///< Center of sub-domain.
66  A da; ///< Length of sub-domain.
67  F f; ///< Sub-function.
68  };
69 
70  using data = std::vector<rec>; ///< Type of data structure for table.
71 
72  private:
73  data dat_; ///< Tabular data.
74 
75  /// Compare argument with record so that right record can be found.
76  static bool comp(A const &a, rec const &r) { return a < r.a; }
77 
78  public:
79  /// Initialize table of sub-domain centers, sub-domain lengths, and
80  /// sub-functions.
81  sparse_table(
82  /// Center \f$ a_0 \f$ of first subdomain.
83  A const &first,
84  /// List of pairs, each containing the length of a sub-domain and a
85  /// sub-function for that sub-domain. The first pair in \a vf is
86  /// interpreted as \f$ (\Delta a_0, f_0) \f$, the second as \f$
87  /// (\Delta a_1, f_1) \f$, etc.
88  std::vector<std::pair<A, F>> vf)
89  : dat_(vf.size())
90  {
91  if (vf.size() == 0) {
92  throw "sparse_table must have at least one entry.";
93  }
94  if (vf[0].first <= A(0)) {
95  throw "Length of sub-domain must be positive.";
96  }
97  dat_[0].a = first;
98  dat_[0].da = vf[0].first;
99  dat_[0].f = vf[0].second;
100  for (unsigned i = 1; i < vf.size(); ++i) {
101  if (vf[i].first <= A(0)) {
102  throw "Length of sub-domain must be positive.";
103  }
104  dat_[i].a = dat_[i - 1].a + 0.5 * (dat_[i - 1].da + dat_[i].da);
105  dat_[i].da = vf[i].first;
106  dat_[i].f = vf[i].second;
107  }
108  }
109 
110  /// Table of sub-domain centers, sub-domain lengths, and sub-functions:
111  /// \f$ (a_0, \Delta a_0, f_0) \f$, \f$ (a_1, \Delta a_1, f_1) \f$, \f$
112  /// \ldots \f$, \f$ (a_{n-1}, \Delta a_{n-1}, f_{n-1}) \f$.
113  data const &dat() const { return dat_; }
114 
115  /// Type of value returned by every sub-function.
116  using R = decltype(F()(A()));
117 
118  /// Find \f$ a_i \f$ whose sub-domain contains \f$ a \f$, and return \f$
119  /// f_i(a - a_i) \f$. If \f$ a < a_0 - \frac{\Delta a_0}{2} \f$, or \f$
120  /// a > a_{n-1} + \frac{\Delta a_{n-1}}{2} \f$, then return 0.
121  ///
122  /// \return \f$ f_i(a - a_i) \f$.
123  R operator()(A const &a /**< Argument to function. */) const
124  {
125  if (a < dat_.begin()->a - 0.5 * dat_.begin()->da ||
126  a > dat_.rbegin()->a + 0.5 * dat_.rbegin()->da) {
127  return R(0);
128  }
129  // In log time, find pointer to first record after argument a.
130  auto p = std::upper_bound(dat_.begin(), dat_.end(), a, comp);
131  if (p == dat_.end()) {
132  p = dat_.rbegin(); // Argument a is after last center.
133  } else if (p->a - a > 0.5 * p->da) {
134  --p; // Argument a is too far from subsequent center.
135  }
136  return p->f(a - p->a);
137  }
138  };
139 }
