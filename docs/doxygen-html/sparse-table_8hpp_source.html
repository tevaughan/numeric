36  /// A [piecewise function](https://en.wikipedia.org/wiki/Piecewise) that, in
37  /// logarithmic time, looks up the sub-function appropriate to the argument.
38  ///
39  /// For any argument \f$a\f$, the corresponding sub-function and its
40  /// sub-domain are found in logarithmic time. The sub-domains are
41  /// contiguous and of variable length. If \f$a\f$ be less than the least
42  /// element of the first subdomain or greater than the greatest element of
43  /// the last subdomain, then there is no corresponding sub-function, and
44  /// zero is returned.
45  ///
46  /// Stored in a table of \f$n\f$ records, the sub-function corresponding to
47  /// \f$a\f$ is \f$f_i\f$, and the center of the corresponding subdomain is
48  /// \f$a_i\f$, where \f$i \in \{0, 1, \ldots, n-1\}\f$. The argument
49  /// \f$a\f$ is passed to \f$f_i\f$, and the value returned by \f$f_i\f$ is
50  /// returned from the lookup.
51  ///
52  /// In the simplest, fastest table, each \f$f_i\f$ ignores its argument and
53  /// returns a constant value.
54  ///
55  /// The infrastructure provided by sparse_table allows for logarithmic-time
56  /// lookup of any kind of sub-function (for example, a cubic interpolant).
57  /// To implement logarithmic-time lookup, an instance of sparse_table stores
58  /// \f$n\f$ triplets
59  /// \f$(a_0, \Delta a_0, f_0)\f$,
60  /// \f$(a_1, \Delta a_1, f_1)\f$,
61  /// \f$\ldots\f$,
62  /// \f$(a_{n-1}, \Delta a_{n-1}, f_{n-1})\f$,
63  /// sorted so that \f$a_0 < a_1 < \cdots < a_{n-1}\f$, where \f$\Delta
64  /// a_i\f$ is the length of the sub-domain whose center is \f$a_i\f$.
65  ///
66  /// The lookup occurs by way of operator()(). When
67  /// \f$a > a_0 - \frac{\Delta a_0}{2}\f$, and
68  /// \f$a < a_{n-1} + \frac{\Delta a_{n-1}}{2}\f$,
69  /// record \f$i\f$ is found so that
70  /// \f$|a - a_i| \leq \frac{{\Delta a}_i}{2}\f$.
71  ///
72  /// operator()() returns \f$f_i(a)\f$.
73  ///
74  /// See dense_table for a type that usually can approximate a given function
75  /// so precisely as sparse_table but with faster lookup. The cost is that
76  /// dense_table requires more storage for the same precision with the same
77  /// sub-function type \a F.
78  ///
79  /// \tparam A Type of sub-function's argument.
80  template <typename A>

129  /// Take a step in combining two tables. This function should be
130  /// called with parameters arranged in a way that identifies which
131  /// piece starts first in the space of arguments of type A.
132  void step(
133  /** Beg of scnd piece. */ A const & b2,
134  /** End of frst piece. */ A const & e1,
135  /** End of scnd piece. */ A const & e2,
136  /** Piece from table on left side of comb. */ rec const &d1,
137  /** Piece from table on rght side of comb. */ rec const &d2,
138  /** Offset into frst piece's table. */ unsigned & i1,
139  /** Offset into scnd piece's table. */ unsigned & i2)
140  {
141  if (e1 < b2) {
142  // d1 ends before d2 starts.
143  ++i1;
144  } else if (e1 < e2) {
145  // d1 ends before d2 ends.
146  d[i].a = 0.5 * (b2 + e1);
147  d[i].da = e1 - b2;
148  d[i].f = cmb(d1.f, d2.f);
149  ++i1;
150  ++i;
151  } else {
152  // d1 ends after d2 ends.
153  d[i].a = d2.a;
154  d[i].da = d2.da;
155  d[i].f = cmb(d1.f, d2.f);
156  ++i2;
157  ++i;
158  }
159  }
160 
161  /// Shrink down to size if necessary. This is meant to be called after
162  /// all of the calls to step() are done.
163  void resize() { d.resize(i); }
164  };

166  /// Combine table with other table, for example by multiplication if \a
167  /// cmb represents the function for multiplication of two expressions.
168  ///
169  /// \return Resutlant table.
170  sparse_table combine(
171  /** Other table. */ sparse_table const &st,
172  /** Combining function. */ std::function<ex(ex, ex)> const &cmb)
173  const
174  {
175  // Initializer for return value. The initial size might be too large
176  // and is reduced if necessary at the end.
177  cmb_data c(dat_.size() + st.dat().size(), cmb);
178  unsigned i1 = 0; // offset into dat_
179  unsigned i2 = 0; // offset into st.dat_
180  while (i1 < dat_.size() && i2 < st.dat().size()) {
181  auto const &d1 = dat_[i1]; // piece from left table
182  auto const &d2 = st.dat()[i2]; // piece from right table
183  auto const hda1 = 0.5 * d1.da;
184  auto const hda2 = 0.5 * d2.da;
185  auto const l1 = d1.a - hda1;
186  auto const l2 = d2.a - hda2;
187  auto const r1 = d1.a + hda1;
188  auto const r2 = d2.a + hda2;
189  if (l1 < l2) {
190  // d1 starts before d2.
191  c.step(l2, r1, r2, d1, d2, i1, i2);
192  } else {
193  // d2 starts before d1.
194  c.step(l1, r2, r1, d1, d2, i2, i1);
195  }
196  }
197  c.resize(); // Shrink down to size if necessary.
198  return sparse_table(std::move(c.d));
199  }

205  /// Initialize table of sub-domain centers, sub-domain lengths, and
206  /// sub-functions.
207  sparse_table(
208  /// Center \f$ a_0 \f$ of first subdomain.
209  A const &a0,
210  /// List of pairs, each containing the length of a sub-domain and a
211  /// sub-function for that sub-domain. The first pair in \a vf is
212  /// interpreted as \f$ (\Delta a_0, f_0) \f$, the second as \f$
213  /// (\Delta a_1, f_1) \f$, etc.
214  std::vector<std::pair<A, ex>> vf)
215  : dat_(vf.size())
216  {
217  if (vf.size() == 0) {
218  throw "sparse_table must have at least one entry.";
219  }
220  std::sort(dat_.begin(), dat_.end(), rcomp);
221  if (vf[0].first <= 0.0 * vf[0].first) {
222  throw "Length of sub-domain must be positive.";
223  }
224  dat_[0].a = a0;
225  dat_[0].da = vf[0].first;
226  dat_[0].f = vf[0].second;
227  for (unsigned i = 1; i < vf.size(); ++i) {
228  if (vf[i].first <= 0.0 * vf[0].first) {
229  throw "Length of sub-domain must be positive.";
230  }
231  dat_[i].da = vf[i].first;
232  dat_[i].f = vf[i].second;
233  dat_[i].a = dat_[i - 1].a + 0.5 * (dat_[i - 1].da + dat_[i].da);
234  }
235  }
236 
237  /// Table of sub-domain centers, sub-domain lengths, and sub-functions:
238  /// \f$ (a_0, \Delta a_0, f_0) \f$,
239  /// \f$ (a_1, \Delta a_1, f_1) \f$,
240  /// \f$ \ldots \f$,
241  /// \f$ (a_{n-1}, \Delta a_{n-1}, f_{n-1}) \f$.
242  data const &dat() const { return dat_; }
243 
244  /// Find \f$a_i\f$ whose sub-domain contains \f$a\f$, and return
245  /// \f$f_i(a)\f$. If
246  /// \f$a < a_0 - \frac{\Delta a_{0}}{2}\f$, or
247  /// \f$a > a_{n-1} + \frac{\Delta a_{n-1}}{2}\f$,
248  /// then return 0.
249  ///
250  /// \return \f$ f_i(a) \f$.
251  ex operator()(/** Argument to function. */ A const &a) const
252  {
253  auto const &frst = *dat_.begin();
254  auto const &last = *dat_.rbegin();
255  if (a < frst.a - 0.5 * frst.da || a > last.a + 0.5 * last.da) {
256  return 0;
257  }
258  // In log time, find pointer to first record after argument a.
259  auto p = std::upper_bound(dat_.begin(), dat_.end(), a, acomp);
260  if (p == dat_.end()) {
261  return last.f.subs(x == a); // Argument a is after last center.
262  } else if (p->a - a > 0.5 * p->da) {
263  --p; // Argument a is too far from subsequent center.
264  }
265  return p->f.subs(x == a);
266  }

268  /// Integral of piece-wise function over all pieces.
269  ex integral() const
270  {
271  ex rv = 0; // Return value.
272  for (auto i : dat_) {
273  rv += GiNaC::integral(x, i.a - 0.5 * i.da, i.a + 0.5 * i.da, i.f)
274  .eval_integ();
275  }
276  return rv.evalf();
277  }

279  /// Integral of piece-wise function over range.
280  ex integral(/** Beg of range. */ A a, /** End of range. */ A b) const
281  {
282  double sign = 1.0;
283  if (a > b) {
284  sign = -1.0;
285  std::swap(a, b);
286  }
287  ex rv = 0; // Return value.
288  if (dat_.size() == 0) {
289  return rv; // There are no pieces over which to integrate.
290  }
291  A const beg = dat_.begin()->a - 0.5 * dat_.begin()->da;
292  A const end = dat_.rbegin()->a + 0.5 * dat_.rbegin()->da;
293  if (a > end) {
294  return rv; // Interval is entirely past the last piece.
295  }
296  if (b < beg) {
297  return rv; // Interval is entirely before the first piece.
298  }
299  if (a < beg) {
300  a = beg; // Interval is partially before the first piece.
301  }
302  if (b > end) {
303  b = end; // Interval is partially after the last piece.
304  }
305  auto pa = std::upper_bound(dat_.begin(), dat_.end(), a, acomp);
306  auto pb = std::upper_bound(dat_.begin(), dat_.end(), b, acomp);
307  if (pa == dat_.end()) {
308  // Beginning of interval is after last center.
309  return (sign * GiNaC::integral(x, a, b, dat_.rbegin()->f))
310  .eval_integ()
311  .evalf();
312  } else if (pa->a - a > 0.5 * pa->da) {
313  --pa; // Beginning of interval is too far from subsequent center.
314  }
315  if (pb == dat_.end() || pb->a - b > 0.5 * pb->da) {
316  // End of interval is after last center or too far from subsequent
317  // center.
318  --pb;
319  }
320  auto const pend = pb + 1;
321  for (auto i = pa; i != pend; ++i) {
322  A aa; // Current beg of integration.
323  A bb; // Current end of integration.
324  if (i == pa) { //
325  aa = a; // Beg of interval.
326  } else { //
327  aa = i->a - 0.5 * i->da; // Beg of piece.
328  } //
329  if (i == pb) { //
330  bb = b; // End of interval.
331  } else { //
332  bb = i->a + 0.5 * i->da; // End of piece.
333  } //
334  // Integral over current piece.
335  rv += GiNaC::integral(x, aa, bb, i->f).eval_integ();
336  }
337  return (sign * rv).evalf();
338  }
