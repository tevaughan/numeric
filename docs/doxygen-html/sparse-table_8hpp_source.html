26  /// A [piecewise function](https://en.wikipedia.org/wiki/Piecewise) that, in
27  /// logarithmic time, looks up the sub-function appropriate to the argument.
28  ///
29  /// For any argument \f$a\f$, the corresponding sub-function and its
30  /// sub-domain are found in logarithmic time. The sub-domains are
31  /// contiguous and of variable length. If \f$a\f$ be less than the least
32  /// element of the first subdomain or greater than the greatest element of
33  /// the last subdomain, then there is no corresponding sub-function, and
34  /// zero is returned.
35  ///
36  /// Stored in a table of \f$n\f$ records, the sub-function corresponding to
37  /// \f$a\f$ is \f$f_i\f$, and the center of the corresponding subdomain is
38  /// \f$a_i\f$, where \f$i \in \{0, 1, \ldots, n-1\}\f$. The argument
39  /// \f$a\f$ is passed to \f$f_i\f$, and the value returned by \f$f_i\f$ is
40  /// returned from the lookup.
41  ///
42  /// In the simplest, fastest table, each \f$f_i\f$ ignores its argument and
43  /// returns a constant value.
44  ///
45  /// The infrastructure provided by sparse_table allows for logarithmic-time
46  /// lookup of any kind of sub-function (for example, a cubic interpolant).
47  /// To implement logarithmic-time lookup, an instance of sparse_table stores
48  /// \f$n\f$ triplets
49  /// \f$(a_0, \Delta a_0, f_0)\f$,
50  /// \f$(a_1, \Delta a_1, f_1)\f$,
51  /// \f$\ldots\f$,
52  /// \f$(a_{n-1}, \Delta a_{n-1}, f_{n-1})\f$,
53  /// sorted so that \f$a_0 < a_1 < \cdots < a_{n-1}\f$, where \f$\Delta
54  /// a_i\f$ is the length of the sub-domain whose center is \f$a_i\f$.
55  ///
56  /// The lookup occurs by way of operator()(). When
57  /// \f$a > a_0 - \frac{\Delta a_0}{2}\f$, and
58  /// \f$a < a_{n-1} + \frac{\Delta a_{n-1}}{2}\f$,
59  /// record \f$i\f$ is found so that
60  /// \f$|a - a_i| \leq \frac{{\Delta a}_i}{2}\f$.
61  ///
62  /// operator()() returns \f$f_i(a)\f$.
63  ///
64  /// See dense_table for a type that usually can approximate a given function
65  /// so precisely as sparse_table but with faster lookup. The cost is that
66  /// dense_table requires more storage for the same precision with the same
67  /// sub-function type \a F.
68  ///
69  /// \tparam A Type of sub-function's argument.
70  template <typename A>
71  class sparse_table : public sparse_table_base

101  /// Initialize table of sub-domain centers, sub-domain lengths, and
102  /// sub-functions.
103  sparse_table(
104  /// Center \f$ a_0 \f$ of first subdomain.
105  A const &a0,
106  /// List of pairs, each containing the length of a sub-domain and a
107  /// sub-function for that sub-domain. The first pair in \a vf is
108  /// interpreted as \f$ (\Delta a_0, f_0) \f$, the second as \f$
109  /// (\Delta a_1, f_1) \f$, etc.
110  std::vector<std::pair<A, GiNaC::ex>> vf)
111  : dat_(vf.size())
112  {
113  if (vf.size() == 0) {
114  throw "sparse_table must have at least one entry.";
115  }
116  std::sort(dat_.begin(), dat_.end(), rcomp);
117  if (vf[0].first <= 0.0 * vf[0].first) {
118  throw "Length of sub-domain must be positive.";
119  }
120  dat_[0].a = a0;
121  dat_[0].da = vf[0].first;
122  dat_[0].f = vf[0].second;
123  for (unsigned i = 1; i < vf.size(); ++i) {
124  if (vf[i].first <= 0.0 * vf[0].first) {
125  throw "Length of sub-domain must be positive.";
126  }
127  dat_[i].da = vf[i].first;
128  dat_[i].f = vf[i].second;
129  dat_[i].a = dat_[i - 1].a + 0.5 * (dat_[i - 1].da + dat_[i].da);
130  }
131  }
132 
133  /// Table of sub-domain centers, sub-domain lengths, and sub-functions:
134  /// \f$ (a_0, \Delta a_0, f_0) \f$,
135  /// \f$ (a_1, \Delta a_1, f_1) \f$,
136  /// \f$ \ldots \f$,
137  /// \f$ (a_{n-1}, \Delta a_{n-1}, f_{n-1}) \f$.
138  data const &dat() const { return dat_; }
139 
140  /// Find \f$a_i\f$ whose sub-domain contains \f$a\f$, and return
141  /// \f$f_i(a)\f$. If
142  /// \f$a < a_0 - \frac{\Delta a_{0}}{2}\f$, or
143  /// \f$a > a_{n-1} + \frac{\Delta a_{n-1}}{2}\f$,
144  /// then return 0.
145  ///
146  /// \return \f$ f_i(a) \f$.
147  GiNaC::ex operator()(/** Argument to function. */ A const &a) const
148  {
149  auto const &frst = *dat_.begin();
150  auto const &last = *dat_.rbegin();
151  if (a < frst.a - 0.5 * frst.da || a > last.a + 0.5 * last.da) {
152  return 0;
153  }
154  // In log time, find pointer to first record after argument a.
155  auto p = std::upper_bound(dat_.begin(), dat_.end(), a, acomp);
156  if (p == dat_.end()) {
157  return last.f.subs(x == a); // Argument a is after last center.
158  } else if (p->a - a > 0.5 * p->da) {
159  --p; // Argument a is too far from subsequent center.
160  }
161  return p->f.subs(x == a);
162  }
163 
164  /// Integral of piece-wise function over all pieces.
165  GiNaC::ex integral() const
166  {
167  GiNaC::ex rv = 0; // Return value.
168  for (auto i : dat_) {
169  rv += GiNaC::integral(x, i.a - 0.5 * i.da, i.a + 0.5 * i.da, i.f);
170  }
171  return rv.evalf();
172  }
173 
174  /// Integral of piece-wise function over range.
175  GiNaC::ex
176  integral(/** Beginning of range. */ A a, /** End of range. */ A b) const
177  {
178  double sign = 1.0;
179  if (a > b) {
180  sign = -1.0;
181  std::swap(a, b);
182  }
183  GiNaC::ex rv = 0; // Return value.
184  if (dat_.size() == 0) {
185  return rv; // There are no pieces over which to integrate.
186  }
187  A const beg = dat_.begin()->a - 0.5 * dat_.begin()->da;
188  A const end = dat_.rbegin()->a + 0.5 * dat_.rbegin()->da;
189  if (a > end) {
190  return rv; // Interval is entirely past the last piece.
191  }
192  if (b < beg) {
193  return rv; // Interval is entirely before the first piece.
194  }
195  if (a < beg) {
196  a = beg; // Interval is partially before the first piece.
197  }
198  if (b > end) {
199  b = end; // Interval is partially after the last piece.
200  }
201  auto pa = std::upper_bound(dat_.begin(), dat_.end(), a, acomp);
202  auto pb = std::upper_bound(dat_.begin(), dat_.end(), b, acomp);
203  if (pa == dat_.end()) {
204  // Beginning of interval is after last center.
205  return (sign * GiNaC::integral(x, a, b, dat_.rbegin()->f)).evalf();
206  } else if (pa->a - a > 0.5 * pa->da) {
207  --pa; // Beginning of interval is too far from subsequent center.
208  }
209  if (pb == dat_.end() || pb->a - b > 0.5 * pb->da) {
210  // End of interval is after last center or too far from subsequent
211  // center.
212  --pb;
213  }
214  auto const pend = pb + 1;
215  for (auto i = pa; i != pend; ++i) {
216  A aa; // Current beg of integration.
217  A bb; // Current end of integration.
218  if (i == pa) { //
219  aa = a; // Beg of interval.
220  } else { //
221  aa = i->a - 0.5 * i->da; // Beg of piece.
222  } //
223  if (i == pb) { //
224  bb = b; // End of interval.
225  } else { //
226  bb = i->a + 0.5 * i->da; // End of piece.
227  } //
228  // Integral over current piece.
229  rv += GiNaC::integral(x, aa, bb, i->f);
230  }
231  return (sign * rv).evalf();
232  }

235  sparse_table operator*(/** Factor. */ GiNaC::ex const &fac) const
236  {

290  /// Multiply table by other table.
291  sparse_table
292  operator*(sparse_table const &st)
293  {
294  // Initializer for return value. The initial size might be too large
295  // and is reduced if necessary at the end.
296  data d(dat_.size() + st.dat().size());
297  unsigned i = 0; // offset into returned array
298  unsigned i1 = 0; // offset into dat_
299  unsigned i2 = 0; // offset into st.dat_
300  while (i1 < dat_.size() && i2 < st.dat().size()) {
301  auto const &d1 = dat_[i1];
302  auto const &d2 = st.dat()[i2];
303  auto const hda1 = 0.5 * d1.da;
304  auto const hda2 = 0.5 * d2.da;
305  auto const l1 = d1.a - hda1;
306  auto const l2 = d2.a - hda2;
307  auto const r1 = d1.a + hda1;
308  auto const r2 = d2.a + hda2;
309  if (l1 < l2) {
310  // d1 starts before d2.
311  if (r1 < l2) {
312  // d1 ends before d2 starts.
313  ++i1;
314  } else if (r1 < r2) {
315  // d1 ends before d2 ends.
316  d[i].a = 0.5 * (l2 + r1);
317  d[i].da = r1 - l2;
318  d[i].f = d1.f * d2.f;
319  ++i1;
320  ++i;
321  } else {
322  // d1 ends after d2 ends.
323  d[i].a = d2.a;
324  d[i].da = d2.da;
325  d[i].f = d1.f * d2.f;
326  ++i2;
327  ++i;
328  }
329  } else {
330  // d2 starts before d1.
331  if (r2 < l1) {
332  // d2 ends before d1 starts.
333  ++i2;
334  } else if (r2 < r1) {
335  // d2 ends before d1 ends.
336  d[i].a = 0.5 * (l1 + r2);
337  d[i].da = r2 - l1;
338  d[i].f = d1.f * d2.f;
339  ++i2;
340  ++i;
341  } else {
342  // d2 ends after d1 ends.
343  d[i].a = d1.a;
344  d[i].da = d1.da;
345  d[i].f = d1.f * d2.f;
346  ++i1;
347  ++i;
348  }
349  }
350  }
351  d.resize(i); // Shrink down to size if necessary.
352  return sparse_table(std::move(d));
353  }
