41  /// A [piecewise function](https://en.wikipedia.org/wiki/Piecewise) that, in
42  /// logarithmic time, looks up the sub-function appropriate to the argument.
43  ///
44  /// For any argument \f$a\f$, the corresponding sub-function and its
45  /// sub-domain are found in logarithmic time. The sub-domains are
46  /// contiguous and of variable length. If \f$a\f$ be less than the least
47  /// element of the first subdomain or greater than the greatest element of
48  /// the last subdomain, then there is no corresponding sub-function, and
49  /// zero is returned.
50  ///
51  /// Stored in a table of \f$n\f$ records, the sub-function corresponding to
52  /// \f$a\f$ is \f$f_i\f$, and the center of the corresponding subdomain is
53  /// \f$a_i\f$, where \f$i \in \{0, 1, \ldots, n-1\}\f$. The argument
54  /// \f$a\f$ is passed to \f$f_i\f$, and the value returned by \f$f_i\f$ is
55  /// returned from the lookup.
56  ///
57  /// In the simplest, fastest table, each \f$f_i\f$ ignores its argument and
58  /// returns a constant value.
59  ///
60  /// The infrastructure provided by sparse_table allows for logarithmic-time
61  /// lookup of any kind of sub-function (for example, a cubic interpolant).
62  /// To implement logarithmic-time lookup, an instance of sparse_table stores
63  /// \f$n\f$ triplets
64  /// \f$(a_0, \Delta a_0, f_0)\f$,
65  /// \f$(a_1, \Delta a_1, f_1)\f$,
66  /// \f$\ldots\f$,
67  /// \f$(a_{n-1}, \Delta a_{n-1}, f_{n-1})\f$,
68  /// sorted so that \f$a_0 < a_1 < \cdots < a_{n-1}\f$, where \f$\Delta
69  /// a_i\f$ is the length of the sub-domain whose center is \f$a_i\f$.
70  ///
71  /// The lookup occurs by way of operator()(). When
72  /// \f$a > a_0 - \frac{\Delta a_0}{2}\f$, and
73  /// \f$a < a_{n-1} + \frac{\Delta a_{n-1}}{2}\f$,
74  /// record \f$i\f$ is found so that
75  /// \f$|a - a_i| \leq \frac{{\Delta a}_i}{2}\f$.
76  ///
77  /// operator()() returns \f$f_i(a)\f$.
78  ///
79  /// See dense_table for a type that usually can approximate a given function
80  /// so precisely as sparse_table but with faster lookup. The cost is that
81  /// dense_table requires more storage for the same precision with the same
82  /// sub-function type \a F.
83  ///
84  /// \tparam A Type of sub-function's argument.
85  template <typename A>
86  class sparse_table : public sparse_table_base

136  /// Initialize table of sub-domain centers, sub-domain lengths, and
137  /// sub-functions.
138  sparse_table(
139  /// Center \f$ a_0 \f$ of first subdomain.
140  A const &a0,
141  /// List of pairs, each containing the length of a sub-domain and a
142  /// sub-function for that sub-domain. The first pair in \a vf is
143  /// interpreted as \f$ (\Delta a_0, f_0) \f$, the second as \f$
144  /// (\Delta a_1, f_1) \f$, etc.
145  std::vector<std::pair<A, GiNaC::ex>> vf)
146  : dat_(vf.size())
147  {
148  if (vf.size() == 0) {
149  throw "sparse_table must have at least one entry.";
150  }
151  std::sort(dat_.begin(), dat_.end(), rcomp);
152  if (vf[0].first <= 0.0 * vf[0].first) {
153  throw "Length of sub-domain must be positive.";
154  }
155  dat_[0].a = a0;
156  dat_[0].da = vf[0].first;
157  dat_[0].f = vf[0].second;
158  for (unsigned i = 1; i < vf.size(); ++i) {
159  if (vf[i].first <= 0.0 * vf[0].first) {
160  throw "Length of sub-domain must be positive.";
161  }
162  dat_[i].da = vf[i].first;
163  dat_[i].f = vf[i].second;
164  dat_[i].a = dat_[i - 1].a + 0.5 * (dat_[i - 1].da + dat_[i].da);
165  }
166  }

168  /// Table of sub-domain centers, sub-domain lengths, and sub-functions:
169  /// \f$ (a_0, \Delta a_0, f_0) \f$,
170  /// \f$ (a_1, \Delta a_1, f_1) \f$,
171  /// \f$ \ldots \f$,
172  /// \f$ (a_{n-1}, \Delta a_{n-1}, f_{n-1}) \f$.
173  data const &dat() const { return dat_; }
174 
175  /// Find \f$a_i\f$ whose sub-domain contains \f$a\f$, and return
176  /// \f$f_i(a)\f$. If
177  /// \f$a < a_0 - \frac{\Delta a_{0}}{2}\f$, or
178  /// \f$a > a_{n-1} + \frac{\Delta a_{n-1}}{2}\f$,
179  /// then return 0.
180  ///
181  /// \return \f$ f_i(a) \f$.
182  GiNaC::ex operator()(/** Argument to function. */ A const &a) const
183  {
184  auto const &frst = *dat_.begin();
185  auto const &last = *dat_.rbegin();
186  if (a < frst.a - 0.5 * frst.da || a > last.a + 0.5 * last.da) {
187  return 0;
188  }
189  // In log time, find pointer to first record after argument a.
190  auto p = std::upper_bound(dat_.begin(), dat_.end(), a, acomp);
191  if (p == dat_.end()) {
192  return last.f.subs(x == a); // Argument a is after last center.
193  } else if (p->a - a > 0.5 * p->da) {
194  --p; // Argument a is too far from subsequent center.
195  }
196  return p->f.subs(x == a);
197  }

199  /// Integral of piece-wise function over all pieces.
200  GiNaC::ex integral() const
201  {
202  GiNaC::ex rv = 0; // Return value.
203  for (auto i : dat_) {
204  rv += GiNaC::integral(x, i.a - 0.5 * i.da, i.a + 0.5 * i.da, i.f);
205  }
206  return rv.evalf();
207  }
208 
209  /// Integral of piece-wise function over range.
210  GiNaC::ex
211  integral(/** Beginning of range. */ A a, /** End of range. */ A b) const
212  {
213  double sign = 1.0;
214  if (a > b) {
215  sign = -1.0;
216  std::swap(a, b);
217  }
218  GiNaC::ex rv = 0; // Return value.
219  if (dat_.size() == 0) {
220  return rv; // There are no pieces over which to integrate.
221  }
222  A const beg = dat_.begin()->a - 0.5 * dat_.begin()->da;
223  A const end = dat_.rbegin()->a + 0.5 * dat_.rbegin()->da;
224  if (a > end) {
225  return rv; // Interval is entirely past the last piece.
226  }
227  if (b < beg) {
228  return rv; // Interval is entirely before the first piece.
229  }
230  if (a < beg) {
231  a = beg; // Interval is partially before the first piece.
232  }
233  if (b > end) {
234  b = end; // Interval is partially after the last piece.
235  }
236  auto pa = std::upper_bound(dat_.begin(), dat_.end(), a, acomp);
237  auto pb = std::upper_bound(dat_.begin(), dat_.end(), b, acomp);
238  if (pa == dat_.end()) {
239  // Beginning of interval is after last center.
240  return (sign * GiNaC::integral(x, a, b, dat_.rbegin()->f)).evalf();
241  } else if (pa->a - a > 0.5 * pa->da) {
242  --pa; // Beginning of interval is too far from subsequent center.
243  }
244  if (pb == dat_.end() || pb->a - b > 0.5 * pb->da) {
245  // End of interval is after last center or too far from subsequent
246  // center.
247  --pb;
248  }
249  auto const pend = pb + 1;
250  for (auto i = pa; i != pend; ++i) {
251  A aa; // Current beg of integration.
252  A bb; // Current end of integration.
253  if (i == pa) { //
254  aa = a; // Beg of interval.
255  } else { //
256  aa = i->a - 0.5 * i->da; // Beg of piece.
257  } //
258  if (i == pb) { //
259  bb = b; // End of interval.
260  } else { //
261  bb = i->a + 0.5 * i->da; // End of piece.
262  } //
263  // Integral over current piece.
264  rv += GiNaC::integral(x, aa, bb, i->f);
265  }
266  return (sign * rv).evalf();
267  }

312  /// Multiply table by other table.
313  sparse_table
314  operator*(sparse_table const &st)
315  {
316  // Initializer for return value. The initial size might be too large
317  // and is reduced if necessary at the end.
318  data d(dat_.size() + st.dat().size());
319  unsigned i = 0; // offset into returned array
320  unsigned i1 = 0; // offset into dat_
321  unsigned i2 = 0; // offset into st.dat_
322  while (i1 < dat_.size() && i2 < st.dat().size()) {
323  auto const &d1 = dat_[i1];
324  auto const &d2 = st.dat()[i2];
325  auto const hda1 = 0.5 * d1.da;
326  auto const hda2 = 0.5 * d2.da;
327  auto const l1 = d1.a - hda1;
328  auto const l2 = d2.a - hda2;
329  auto const r1 = d1.a + hda1;
330  auto const r2 = d2.a + hda2;
331  if (l1 < l2) {
332  // d1 starts before d2.
333  if (r1 < l2) {
334  // d1 ends before d2 starts.
335  ++i1;
336  } else if (r1 < r2) {
337  // d1 ends before d2 ends.
338  d[i].a = 0.5 * (l2 + r1);
339  d[i].da = r1 - l2;
340  d[i].f = d1.f * d2.f;
341  ++i1;
342  ++i;
343  } else {
344  // d1 ends after d2 ends.
345  d[i].a = d2.a;
346  d[i].da = d2.da;
347  d[i].f = d1.f * d2.f;
348  ++i2;
349  ++i;
350  }
351  } else {
352  // d2 starts before d1.
353  if (r2 < l1) {
354  // d2 ends before d1 starts.
355  ++i2;
356  } else if (r2 < r1) {
357  // d2 ends before d1 ends.
358  d[i].a = 0.5 * (l1 + r2);
359  d[i].da = r2 - l1;
360  d[i].f = d1.f * d2.f;
361  ++i2;
362  ++i;
363  } else {
364  // d2 ends after d1 ends.
365  d[i].a = d1.a;
366  d[i].da = d1.da;
367  d[i].f = d1.f * d2.f;
368  ++i1;
369  ++i;
370  }
371  }
372  }
373  d.resize(i); // Shrink down to size if necessary.
374  return sparse_table(std::move(d));
375  }
