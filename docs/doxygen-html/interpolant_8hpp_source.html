56  /// Extract a set of points from a space-delimited ASCII file. Each line
57  /// of the file must consist either
58  ///
59  /// - of only white space, optionally followed by the '#' character and a
60  /// subsequent comment or
61  ///
62  /// - of white space followed by at least two space-delimited
63  /// floating-point numbers (after which everything else on the line is
64  /// ignored).
65  ///
66  /// \tparam X Type of first column, representing X coordinate.
67  /// \tparam Y Type of second column, representing Y coordinate.
68  template <typename X, typename Y>
69  ilist<X, Y> get_points(
70  /** Name of ASCII file. */ std::string file,
71  /** Unit to multiply against first column. */ X xu,
72  /** Unit to multiply against secnd column. */ Y yu)
73  {
74  std::ifstream is(file);
75  if (!is) {
76  throw "Failed to open '" + file + "'.";
77  }
78  std::string line;
79  ilist<X, Y> points;
80  while (getline(is, line)) {
81  size_t const p = line.find_first_not_of(" \f\n\r\t");
82  if (p == std::string::npos || line[p] == '#') {
83  continue;
84  }
85  points.push_back(get_point(line, xu, yu));
86  }
87  auto comp = [](ipoint<X, Y> const &a, ipoint<X, Y> const &b) {
88  return a.first < b.first;

94  /// Return the list of midpoints, each between a subsequent pair of
95  /// points in a sorted list.
96  ///
97  /// \tparam V Type of vector of points.
98  /// \return Midpoints sorted by x coordinate.
99  template <typename V>
100  V midpoints(/** Points sorted by x coordinate. */ V const &v)
101  {
102  if (v.size() < 2) {
103  return V();
104  }
105  V rv(v.size() - 1); // Return value.
106  for (unsigned j = 1; j < v.size(); ++j) {
107  unsigned const i = j - 1;
108  auto const & xi = v[i].first, &yi = v[i].second;
109  auto const & xj = v[j].first, &yj = v[j].second;
110  rv[i] = {0.5 * (xi + xj), 0.5 * (yi + yj)};
111  }
112  return rv;
113  }

116  /// Construct a (\ref sparse_table) piecewise-constant interpolant from the
117  /// first two space-delimited columns in an ASCII file. Each line of the
118  /// file must consist either
119  ///
120  /// - of only white space, optionally followed by the '#' character and a
121  /// subsequent comment or
122  ///
123  /// - of white space followed by at least two space-delimited floating-point
124  /// numbers (after which everything else on the line is ignored).
125  ///
126  /// \tparam X Type of first column, representing the x coordinate.
127  /// \tparam Y Type of second column, representing the Y coordinate.
128  template <typename X = double, typename Y = double>
129  sparse_table<X, cpoly<0, X, Y>> make_const_interp(
130  /** Name of ASCII file. */ std::string file,
131  /** Unit multiplying first col. */ X const &xu = 1,
132  /** Unit multiplying secnd col. */ Y const &yu = 1)
133  {
134  using V = ilist<X, Y>;
135  using C = cpoly<0, X, Y>;
136  V const cp = util::get_points(file, xu, yu); // control points
137  if (cp.size() < 1) {
138  throw "Must have at least one control point.";
139  }
140  // For linear interpolation, each subdomain is just the x region between
141  // subsequent control points. For constant interpolation, however, each
142  // subdomain is the x region between subsequent *midpoints* between
143  // control points, each between a pair of control points. However, in
144  // general only each of the first and last points lies in the center of
145  // its subdomain.
146  V const cpmid = util::midpoints(cp); // control midpoints
147  X const a0 = cp[0].first;
148  std::vector<std::pair<X, C>> vf(cp.size());
149  vf[0].first = 2.0 * (cpmid[0].first - cp[0].first);
150  vf[0].second = cp[0].second;
151  for (unsigned i = 1; i < cpmid.size(); ++i) {
152  vf[i].first = cpmid[i].first - cpmid[i - 1].first;
153  vf[i].second = cp[i].second;
154  }
155  unsigned const i = cpmid.size();
156  vf[i].first = 2.0 * (cp[i].first - cpmid[i - 1].first);
157  vf[i].second = cp[i].second;
158  return sparse_table<X, C>(a0, move(vf));
159  }

161  /// Construct a (\ref sparse_table) piecewise-linear interpolant from a set
162  /// of ordered pairs.
163  ///
164  /// \tparam X Type of first element of each ordered pair.
165  /// \tparam Y Type of second element of each ordered pair.
166  template <typename X = double, typename Y = double>
167  sparse_table<X, cpoly<1, X, Y>> make_linear_interp(ilist<X, Y> const &cp)
168  {
169  using C = cpoly<1, X, Y>;
170  if (cp.size() == 0) {
171  return sparse_table<X, C>();
172  } else if (cp.size() == 1) {
173  throw "Must have at least two control points.";
174  }
175  // For linear interpolation, each subdomain is just the x region between
176  // subsequent control points.
177  X const a0 = 0.5 * (cp[0].first + cp[1].first);
178  unsigned const ndeltas = cp.size() - 1;
179  std::vector<std::pair<X, C>> vf(ndeltas);
180  for (unsigned i = 0; i < ndeltas; ++i) {
181  unsigned const j = i + 1;
182  X const delta = cp[j].first - cp[i].first;
183  Y const c0 = 0.5 * (cp[i].second + cp[j].second);
184  auto const c1 = (cp[j].second - cp[i].second) / delta;
185  vf[i].first = delta;
186  vf[i].second.template set_coef<0>(c0);
187  vf[i].second.template set_coef<1>(c1);
188  }
189  return sparse_table<X, C>(a0, move(vf));
190  }

295  /// Combine the present instance I1 with another interpolant I2 such that
296  /// there is a point in the combined interpolant I3 at every unique x
297  /// coordinate across I1 and I2. Each new y coordinate is the
298  /// combination of a y coordinate, in either I1 or I2, and the
299  /// corresponding interpolated y coordinate in the other. The
300  /// combination is represented by a passed-in function.
301  ///
302  /// combine() is used to support multiplication and division of
303  /// interpolants.
304  ///
305  /// \tparam OD Type of y coordinate in other interpolant.
306  /// \tparam F Type of combining function.
307  /// \param oint Reference to other interpolant.
308  /// \param f Combining function.
309  /// \return Combined interpolant.
310  template <typename OD, typename F>
311  combined<OD, F> combine(interpolant<I, OD> const &oint, F f) const
312  {
313  using CD = decltype(f(D(), OD()));
314  ilist<I, CD> pl;
315  ilist<I, D> const & l1 = d_;
316  ilist<I, OD> const &l2 = oint.d_;
317  unsigned const s1 = l1.size();
318  unsigned const s2 = l2.size();
319  unsigned n1 = 0;
320  unsigned n2 = 0;
321  I ind;
322  if (s1 == 0 || s2 == 0) {
323  return pl;
324  }
325  while (n1 < s1 || n2 < s2) {
326  if (n1 < s1 && n2 < s2) {
327  I const i1 = l1[n1].first;
328  I const i2 = l2[n2].first;
329  D const d1 = l1[n1].second;
330  OD const d2 = l2[n2].second;
331  if (i1 < i2) {
332  pl.push_back({i1, f(d1, oint(i1))});
333  ++n1;
334  } else {
335  pl.push_back({i2, f((*this)(i2), d2)});
336  ++n2;
337  }
338  } else if (n1 < s1) {
339  I const i1 = l1[n1].first;
340  D const d1 = l1[n1].second;
341  pl.push_back({i1, f(d1, oint(i1))});
342  ++n1;
343  } else {
344  I const i2 = l2[n2].first;
345  OD const d2 = l2[n2].second;
346  pl.push_back({i2, f((*this)(i2), d2)});
347  ++n2;
348  }
349  }
350  return pl;
351  }

354  void init_from_stack(subinterval_stack<I, D> s)
355  {
356  using interval = interval<I, D>;
357  // Default ordering of stack is not what we want here.
358  auto comp = [](interval const &i1, interval const &i2) {
359  return i1.a < i2.a;

361  // Work from right-most to left-most subinterval.
362  std::sort(s.begin(), s.end(), comp);
363  while (s.size() > 1) {

377  /// Initialize from a continuous function and from an interval on its
378  /// domain. The initial number of evenly spaced samples should be
379  /// sufficient to allow recursive subdivision to produce an interpolant
380  /// with the desired fractional tolerance as an approximation of the
381  /// function.
382  ///
383  /// \tparam A1 Type of value indicating left edge of domain. An
384  /// instance of A1 must be convertible to type I.
385  ///
386  /// \tparam A2 Type of value indicating right edge of domain. An
387  /// instance of A2 must be convertible to type I.
388  ///
389  /// \param f Function to approximate via interpolation.
390  /// \param aa Left edge of domain.
391  /// \param bb Right edge of domain.
392  /// \param t Fractional tolerance of approximation.
393  /// \param n Initial number of evenly spaced samples of function.
394  template <typename A1, typename A2>
395  void
396  init(std::function<D(I)> f, A1 aa, A2 bb, double t = 1.0E-06,
397  unsigned n = 16)
398  {
399  double constexpr eps = std::numeric_limits<double>::epsilon();
400  double constexpr min_tol = 1000.0 * eps;
401  double tol = t;
402  if (tol <= 0.0) {
403  throw "tolerance not positive";
404  } else if (tol < min_tol) {
405  tol = min_tol;
406  }
407  double sign = 1.0;
408  I a = aa;
409  I b = bb;
410  if (a > b) {
411  std::swap(a, b);
412  sign = -1.0;
413  }
414  subinterval_stack<I, D> s(n, a, b, f);
415  init_from_stack(s); // Add initial n points to d_.
416  integral_stats<A> stats(0.0 * aa * f(aa));
417  while (s.size()) {
418  using interval = interval<I, D>;
419  interval const r = *s.rbegin();
420  s.pop_back();
421  I const midp = 0.5 * (r.a + r.b); // midpoint of interval
422  D const fmid = f(midp); // function value at midpoint
423  I const len = r.b - r.a; // length of interval
424  D const mean = 0.5 * (r.fa + r.fb); // mean of function values
425  D const rmean = 0.5 * (mean + fmid); // refined mean
426  D const u0 = fabs(rmean);
427  D const u1 = fabs(mean - rmean);
428  D const u2 = fabs(mean + rmean);
429  D const u3 = u0 * tol;
430  A const ds = rmean * len;
431  if (u1 <= u3) {
432  // Estimated error sufficiently small. Stop refining estimate.
433  stats.add(ds, u1 * len);
434  d_.push_back({midp, fmid});
435  } else if (u1 <= u2 * tol) {
436  // Calculated error too small. Stop refining estimate.
437  stats.add(ds, u1 * len);
438  d_.push_back({midp, fmid});
439  } else if (len <= fabs(midp) * tol) {
440  // Length of interval too small. Stop refining estimate.
441  stats.add(ds, u1 * len);
442  d_.push_back({midp, fmid});
443  } else if (fabs(ds) <= fabs(stats.area()) * tol) {
444  // Increment to integral too small. Stop refining estimate.
445  stats.add(ds, u1 * len);
446  d_.push_back({midp, fmid});
447  } else {
448  // Continue subdividing.
449  s.push_back(interval{r.a, midp, r.fa, fmid});
450  s.push_back(interval{midp, r.b, fmid, r.fb});
451  }
452  }
453  A const farea = fabs(stats.area());
454  A const sigma = stats.stdev(); // statistical error
455  A const derr = farea * t; // desired error
456  A const rerr = farea * eps; // round-off error
457  A eerr; // greater of statistical and round-off error
458  if (sigma < rerr) {
459  eerr = rerr;
460  } else {
461  eerr = sigma;
462  }
463  if (eerr > derr) {
464  std::cerr << "integral: WARNING: Estimated error " << eerr / farea
465  << " is greater than tolerance " << t << "."
466  << std::endl;
467  }
468  integral_ = sign * stats.area();
469  sort();
470  }

567  /// Interpolate.
568  /// \param x Value of independent variable.
569  /// \return Interpolated value of dependent variable.
570  D operator()(I x) const
571  {
572  if (d_.size() == 0) {
573  throw "interpolating on empty interpolant";
574  }
575  if (x <= d_.begin()->first) {
576  return d_.begin()->second;
577  }
578  if (x >= d_.rbegin()->first) {
579  return d_.rbegin()->second;
580  }
581  using namespace std;
582  point const xp{x, D()}; // Dummy y-coord used for search.
583  auto const j = upper_bound(d_.begin(), d_.end(), xp, cmpr_pts);
584  auto const i = j - 1;
585  I const & xi = i->first;
586  I const & xj = j->first;
587  D const & yi = i->second;
588  D const & yj = j->second;
589  return yi + (yj - yi) * ((x - xi) / (xj - xi));

592  /// Integrate interpolant.
593  /// \param x1 Beginning of interval of integration.
594  /// \param x2 End of interval of integration.
595  /// \return Numerical result of definite integral.
596  A integral(I x1, I x2) const
597  {
598  if (d_.size() == 0) {
599  throw "integrating on empty interpolant";
600  }
601  double sign = 1.0;
602  if (x1 > x2) {
603  std::swap(x1, x2);
604  sign = -1.0;
605  }
606  using R = decltype(I() * D());
607  R sum(0.0 * x1 * d_[0].second);
608  point const x1p{x1, D()}; // Dummy y-coord for search.
609  auto i = upper_bound(d_.begin(), d_.end(), x1p, cmpr_pts);
610  if (i == d_.end()) {
611  // We need only calculate one piece to the right of all points.
612  return sign * (x2 - x1) * d_.rbegin()->second;
613  } else {
614  // Handle first piece up to x coord of stored point.
615  D const y1 = (*this)(x1);
616  if (x2 <= i->first) {
617  // We need calculate only one piece.
618  return 0.5 * sign * (x2 - x1) * (y1 + (*this)(x2));
619  }
620  sum += 0.5 * (i->first - x1) * (y1 + i->second);
621  }
622  auto j = i + 1;
623  while (j != d_.end() && x2 >= j->first) {
624  // Handle piece between current point and next point.
625  sum += 0.5 * (j->first - i->first) * (i->second + j->second);
626  ++j;
627  i = j - 1;
628  }
629  // Handle last piece.
630  sum += 0.5 * (x2 - i->first) * (i->second + (*this)(x2));
631  return sign * sum;
632  }
