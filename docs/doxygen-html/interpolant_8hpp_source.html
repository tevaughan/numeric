142  /// Combine the present instance I1 with another interpolant I2 such that
143  /// there is a point in the combined interpolant I3 at every unique x
144  /// coordinate across I1 and I2. Each new y coordinate is the
145  /// combination of a y coordinate, in either I1 or I2, and the
146  /// corresponding interpolated y coordinate in the other. The
147  /// combination is represented by a passed-in function.
148  ///
149  /// combine() is used to support multiplication and division of
150  /// interpolants.
151  ///
152  /// \tparam OD Type of y coordinate in other interpolant.
153  /// \param oint Reference to other interpolant.
154  /// \param f Combining function.
155  /// \return Combined interpolant.
156  template <typename OD>
157  auto combine(interpolant<I, OD> const &oint,
158  std::function<decltype(D() * OD())(D, OD)> f) const
159  -> interpolant<I, decltype(D() * OD())>
160  {
161  using PD = decltype(D() * OD());
162  ilist<I, PD> pl;
163  ilist<I, D> const &l1 = d_;
164  ilist<I, OD> const &l2 = oint.d_;
165  unsigned const s1 = l1.size();
166  unsigned const s2 = l2.size();
167  unsigned n1 = 0;
168  unsigned n2 = 0;
169  I ind;
170  if (s1 == 0 || s2 == 0) {
171  return pl;
172  }
173  while (n1 < s1 || n2 < s2) {
174  if (n1 < s1 && n2 < s2) {
175  I const i1 = l1[n1].first;
176  I const i2 = l2[n2].first;
177  D const d1 = l1[n1].second;
178  OD const d2 = l2[n2].second;
179  if (i1 < i2) {
180  pl.push_back({i1, f(d1, oint(i1))});
181  ++n1;
182  } else {
183  pl.push_back({i2, f((*this)(i2), d2)});
184  ++n2;
185  }
186  } else if (n1 < s1) {
187  I const i1 = l1[n1].first;
188  D const d1 = l1[n1].second;
189  pl.push_back({i1, f(d1, oint(i1))});
190  ++n1;
191  } else {
192  I const i2 = l2[n2].first;
193  OD const d2 = l2[n2].second;
194  pl.push_back({i2, f((*this)(i2), d2)});
195  ++n2;
196  }
197  }
198  return pl;
199  }

201  // Initialize control points from copy of initial subinterval_stack.
202  void init_from_stack(subinterval_stack<I, D> s)
203  {
204  using interval = interval<I, D>;
205  // Default ordering of stack is not what we want here.
206  auto comp = [](interval const &i1, interval const &i2) {
207  return i1.a < i2.a;
208  };
209  // Work from right-most to left-most subinterval.
210  std::sort(s.begin(), s.end(), comp);
211  while (s.size() > 1) {
212  // Push right-most end of every subinterval except for left-most
213  // subinterval.
214  interval const r = *s.rbegin();
215  s.pop_back();
216  d_.push_back({r.b, r.fb});
217  }
218  // For left-most subinterval, push both ends.
219  interval const r = *s.rbegin();
220  s.pop_back();
221  d_.push_back({r.b, r.fb});
222  d_.push_back({r.a, r.fa});

225  /// Initialize from a continuous function and from an interval on its
226  /// domain. The initial number of evenly spaced samples should be
227  /// sufficient to allow recursive subdivision to produce an interpolant
228  /// with the desired fractional tolerance as an approximation of the
229  /// function.
230  ///
231  /// \tparam A1 Type of value indicating left edge of domain. An
232  /// instance of A1 must be convertible to type I.
233  ///
234  /// \tparam A2 Type of value indicating right edge of domain. An
235  /// instance of A2 must be convertible to type I.
236  ///
237  /// \param f Function to approximate via interpolation.
238  /// \param aa Left edge of domain.
239  /// \param bb Right edge of domain.
240  /// \param t Fractional tolerance of approximation.
241  /// \param n Initial number of evenly spaced samples of function.
242  template <typename A1, typename A2>
243  void init(std::function<D(I)> f, A1 aa, A2 bb, double t = 1.0E-06,
244  unsigned n = 16)
245  {
246  double constexpr eps = std::numeric_limits<double>::epsilon();
247  double constexpr min_tol = 1000.0 * eps;
248  double tol = t;
249  if (tol <= 0.0) {
250  throw "tolerance not positive";
251  } else if (tol < min_tol) {
252  tol = min_tol;
253  }
254  double sign = 1.0;
255  I a = aa;
256  I b = bb;
257  if (a > b) {
258  std::swap(a, b);
259  sign = -1.0;
260  }
261  subinterval_stack<I, D> s(n, a, b, f);
262  init_from_stack(s); // Add initial n points to d_.
263  integral_stats<A> stats;
264  while (s.size()) {
265  using interval = interval<I, D>;
266  interval const r = *s.rbegin();
267  s.pop_back();
268  I const midp = 0.5 * (r.a + r.b); // midpoint of interval
269  D const fmid = f(midp); // function value at midpoint
270  I const len = r.b - r.a; // length of interval
271  D const mean = 0.5 * (r.fa + r.fb); // mean of function values
272  D const rmean = 0.5 * (mean + fmid); // refined mean
273  D const u0 = fabs(rmean);
274  D const u1 = fabs(mean - rmean);
275  D const u2 = fabs(mean + rmean);
276  D const u3 = u0 * tol;
277  A const ds = rmean * len;
278  if (u1 <= u3) {
279  // Estimated error sufficiently small. Stop refining estimate.
280  stats.add(ds, u1 * len);
281  d_.push_back({midp, fmid});
282  } else if (u1 <= u2 * tol) {
283  // Calculated error too small. Stop refining estimate.
284  stats.add(ds, u1 * len);
285  d_.push_back({midp, fmid});
286  } else if (len <= fabs(midp) * tol) {
287  // Length of interval too small. Stop refining estimate.
288  stats.add(ds, u1 * len);
289  d_.push_back({midp, fmid});
290  } else if (fabs(ds) <= fabs(stats.area()) * tol) {
291  // Increment to integral too small. Stop refining estimate.
292  stats.add(ds, u1 * len);
293  d_.push_back({midp, fmid});
294  } else {
295  // Continue subdividing.
296  s.push_back(interval{r.a, midp, r.fa, fmid});
297  s.push_back(interval{midp, r.b, fmid, r.fb});
298  }
299  }
300  A const farea = fabs(stats.area());
301  A const sigma = stats.stdev(); // statistical error
302  A const nres = A(1.0) + farea; // normalized result
303  A const derr = nres * t; // desired error
304  A const rerr = nres * eps; // round-off error
305  A eerr; // greater of statistical and round-off error
306  if (sigma < rerr) {
307  eerr = rerr;
308  } else {
309  eerr = sigma;
310  }
311  if (eerr > derr) {
312  std::cerr << "integral: WARNING: Estimated error " << eerr / nres
313  << " is greater than tolerance " << t << "."
314  << std::endl;
315  }
316  integral_ = sign * stats.area();
317  sort();
318  }

411  /// Interpolate.
412  /// \param x Value of independent variable.
413  /// \return Interpolated value of dependent variable.
414  D operator()(I x) const
415  {
416  if (d_.size() == 0) {
417  throw "interpolating on empty interpolant";
418  }
419  if (x <= d_.begin()->first) {
420  return d_.begin()->second;
421  }
422  if (x >= d_.rbegin()->first) {
423  return d_.rbegin()->second;
424  }
425  using namespace std;
426  point const xp{x, D()}; // Dummy y-coord used for search.
427  auto const j = upper_bound(d_.begin(), d_.end(), xp, cmpr_pts);
428  auto const i = j - 1;
429  I const &xi = i->first;
430  I const &xj = j->first;
431  D const &yi = i->second;
432  D const &yj = j->second;
433  return yi + (yj - yi) * ((x - xi) / (xj - xi));
434  }

436  /// Integrate interpolant.
437  /// \param x1 Beginning of interval of integration.
438  /// \param x2 End of interval of integration.
439  /// \return Numerical result of definite integral.
440  A integral(I x1, I x2) const
441  {
442  if (d_.size() == 0) {
443  throw "integrating on empty interpolant";
444  }
445  double sign = 1.0;
446  if (x1 > x2) {
447  std::swap(x1, x2);
448  sign = -1.0;
449  }
450  using R = decltype(I() * D());
451  R sum(0);
452  point const x1p{x1, D()}; // Dummy y-coord for search.
453  auto i = upper_bound(d_.begin(), d_.end(), x1p, cmpr_pts);
454  if (i == d_.end()) {
455  // We need only calculate one piece to the right of all points.
456  return sign * (x2 - x1) * d_.rbegin()->second;
457  } else {
458  // Handle first piece up to x coord of stored point.
459  D const y1 = (*this)(x1);
460  if (x2 <= i->first) {
461  // We need calculate only one piece.
462  return 0.5 * sign * (x2 - x1) * (y1 + (*this)(x2));
463  }
464  sum += 0.5 * (i->first - x1) * (y1 + i->second);
465  }
466  auto j = i + 1;
467  while (j != d_.end() && x2 >= j->first) {
468  // Handle piece between current point and next point.
469  sum += 0.5 * (j->first - i->first) * (i->second + j->second);
470  ++j;
471  i = j - 1;
472  }
473  // Handle last piece.
474  sum += 0.5 * (x2 - i->first) * (i->second + (*this)(x2));
475  return sign * sum;
476  }
