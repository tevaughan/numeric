29  /// Construct a (\ref sparse_table) piecewise-constant interpolant from the
30  /// first two space-delimited columns in an ASCII file. Each line of the
31  /// file must consist either
32  ///
33  /// - of only white space, optionally followed by the '#' character and a
34  /// subsequent comment or
35  ///
36  /// - of white space followed by at least two space-delimited floating-point
37  /// numbers (after which everything else on the line is ignored).
38  ///
39  /// \tparam X Type of first column, representing the x coordinate.
40  /// \tparam Y Type of second column, representing the Y coordinate.
41  template <typename X = double, typename Y = double>
42  sparse_table<X> make_const_interp(
43  /** Name of ASCII file. */ std::string file,
44  /** Unit multiplying first col. */ X const &xu = 1,
45  /** Unit multiplying secnd col. */ Y const &yu = 1)
46  {
47  using V = ilist<X, Y>;
48  V const cp = get_points(file, xu, yu); // control points
49  if (cp.size() < 1) {
50  throw "Must have at least one control point.";
51  }
52  // For linear interpolation, each subdomain is just the x region between
53  // subsequent control points. For constant interpolation, however, each
54  // subdomain is the x region between subsequent *midpoints* between
55  // control points, each between a pair of control points. However, in
56  // general only each of the first and last points lies in the center of
57  // its subdomain.
58  V const cpmid = midpoints(cp); // control midpoints
59  X const a0 = cp[0].first;
60  std::vector<std::pair<X, GiNaC::ex>> vf(cp.size());
61  vf[0].first = 2.0 * (cpmid[0].first - cp[0].first);
62  vf[0].second = cp[0].second;
63  for (unsigned i = 1; i < cpmid.size(); ++i) {
64  vf[i].first = cpmid[i].first - cpmid[i - 1].first;
65  vf[i].second = cp[i].second;
66  }
67  unsigned const i = cpmid.size();
68  vf[i].first = 2.0 * (cp[i].first - cpmid[i - 1].first);
69  vf[i].second = cp[i].second;
70  return sparse_table<X>(a0, move(vf));
71  }

73  /// Construct a (\ref sparse_table) piecewise-linear interpolant from a set
74  /// of ordered pairs.
75  ///
76  /// \tparam X Type of first element of each ordered pair.
77  /// \tparam Y Type of second element of each ordered pair.
78  template <typename X = double, typename Y = double>
79  sparse_table<X> make_linear_interp(ilist<X, Y> cp)
80  {
81  if (cp.size() == 0) {
82  return sparse_table<X>();
83  } else if (cp.size() == 1) {
84  throw "Must have at least two control points.";
85  }
86  std::sort(cp.begin(), cp.end());
87  // For linear interpolation, each subdomain is just the x region between
88  // subsequent control points.
89  X const a0 = 0.5 * (cp[0].first + cp[1].first);
90  unsigned const ndeltas = cp.size() - 1;
91  using namespace std;
92  vector<pair<X, GiNaC::ex>> vf(ndeltas);
93  for (unsigned i = 0; i < ndeltas; ++i) {
94  unsigned const j = i + 1;
95  X const & x1 = cp[i].first;
96  X const & x2 = cp[j].first;
97  Y const & y1 = cp[i].second;
98  Y const & y2 = cp[j].second;
99  X const dx = x2 - x1;
100  auto const c1 = (y2 - y1) / dx;
101  Y const c0 = y1 - c1 * x1;
102  vf[i].first = dx;
103  vf[i].second = c0 + c1 * sparse_table<X>::x;
104  }
105  return sparse_table<X>(a0, move(vf));
106  }

133  /// Initialize control points from copy of initial subinterval_stack.
134  /// \tparam X Type of independent variable.
135  /// \tparam Y Type of dependent variable.
136  template <typename X, typename Y>
137  void init_from_stack(
138  /** Copy of stack. */ subinterval_stack<X, Y> s,
139  /** Initially empty list of points. */ ilist<X, Y> & d)
140  {
141  using interval = interval<X, Y>;
142  // Default ordering of stack is not what we want here.
143  auto comp = [](interval const &i1, interval const &i2) {
144  return i1.a < i2.a;
145  };
146  // Work from right-most to left-most subinterval.
147  std::sort(s.begin(), s.end(), comp);
148  while (s.size() > 1) {
149  // Push right-most end of every subinterval except for left-most
150  // subinterval.
151  interval const r = *s.rbegin();
152  s.pop_back();
153  d.push_back({r.b, r.fb});
154  }
155  // For left-most subinterval, push both ends.
156  interval const r = *s.rbegin();
157  s.pop_back();
158  d.push_back({r.b, r.fb});
159  d.push_back({r.a, r.fa});
160  }

162  /// Construct a (\ref sparse_table) piecewise-linear interpolant for a
163  /// continuous function over the specified interval of its domain. The
164  /// initial number of evenly spaced samples should be sufficient to allow
165  /// recursive subdivision to produce an interpolant with the desired
166  /// fractional error tolerance. The numerical integral of the function is
167  /// computed as a by-product and---if the user supply a pointer in the final
168  /// argument---is then stored at the location indicated by the supplied
169  /// pointer.
170  ///
171  /// \tparam X Type of independent variable.
172  /// \tparam Y Type of dependent variable.
173  /// \param f Function to approximate via interpolation.
174  /// \param aa Left edge of domain.
175  /// \param bb Right edge of domain.
176  /// \param t Fractional tolerance of approximation.
177  /// \param n Initial number of evenly spaced samples of function.
178  /// \param i If non-null, pointer to storage integral.
179  template <typename X, typename Y>
180  sparse_table<X> make_linear_interp(
181  std::function<Y(X)> f, X aa, X bb, double t = 1.0E-06,
182  unsigned n = 16, decltype(X() * Y()) *i = nullptr)
183  {
184  double constexpr eps = std::numeric_limits<double>::epsilon();
185  double constexpr min_tol = 1000.0 * eps;
186  double tol = t;
187  if (tol <= 0.0) {
188  throw "tolerance not positive";
189  } else if (tol < min_tol) {
190  tol = min_tol;
191  }
192  double sign = 1.0;
193  if (aa > bb) {
194  std::swap(aa, bb);
195  sign = -1.0;
196  }
197  subinterval_stack<X, Y> s(n, aa, bb, f); // Stack of intervals.
198  ilist<X, Y> d; // Control points.
199  init_from_stack(s, d); // Add initial n points to d.
200  using A = decltype(X() * Y());
201  integral_stats<A> stats(0.0 * aa * f(aa));
202  while (s.size()) {
203  using interval = interval<X, Y>;
204  interval const r = *s.rbegin();
205  s.pop_back();
206  X const midp = 0.5 * (r.a + r.b); // midpoint of interval
207  Y const fmid = f(midp); // function value at midpoint
208  X const len = r.b - r.a; // length of interval
209  Y const mean = 0.5 * (r.fa + r.fb); // mean of function values
210  Y const rmean = 0.5 * (mean + fmid); // refined mean
211  Y const u0 = fabs(rmean);
212  Y const u1 = fabs(mean - rmean);
213  Y const u2 = fabs(mean + rmean);
214  Y const u3 = u0 * tol;
215  A const ds = rmean * len;
216  if (u1 <= u3) {
217  // Estimated error sufficiently small. Stop refining estimate.
218  stats.add(ds, u1 * len);
219  d.push_back({midp, fmid});
220  } else if (u1 <= u2 * tol) {
221  // Calculated error too small. Stop refining estimate.
222  stats.add(ds, u1 * len);
223  d.push_back({midp, fmid});
224  } else if (len <= fabs(midp) * tol) {
225  // Length of interval too small. Stop refining estimate.
226  stats.add(ds, u1 * len);
227  d.push_back({midp, fmid});
228  } else if (fabs(ds) <= fabs(stats.area()) * tol) {
229  // Increment to integral too small. Stop refining estimate.
230  stats.add(ds, u1 * len);
231  d.push_back({midp, fmid});
232  } else {
233  // Continue subdividing.
234  s.push_back(interval{r.a, midp, r.fa, fmid});
235  s.push_back(interval{midp, r.b, fmid, r.fb});
236  }
237  }
238  A const farea = fabs(stats.area());
239  A const sigma = stats.stdev(); // statistical error
240  A const derr = farea * t; // desired error
241  A const rerr = farea * eps; // round-off error
242  A eerr; // greater of statistical and round-off error
243  if (sigma < rerr) {
244  eerr = rerr;
245  } else {
246  eerr = sigma;
247  }
248  if (eerr > derr) {
249  std::cerr << "integral: WARNING: Estimated error " << eerr / farea
250  << " is greater than tolerance " << t << "." << std::endl;
251  }
252  if (i) {
253  *i = sign * stats.area();
254  }
255  return make_linear_interp(d);
256  }
